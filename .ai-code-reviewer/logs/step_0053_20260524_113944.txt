fn expand_number(spec: &str) -> Option<i64> {
    let spec = spec.trim();
    if spec.is_empty() {
        return None;
    }
    let (digits, radix) = if spec.starts_with("0x") || spec.starts_with("0X") {
        (&spec[2..], 16)
    } else if spec.starts_with('0') && spec.len() > 1 {
        (spec, 8)
    } else {
        (spec, 10)
    };
    if digits.is_empty() {
        return None;
    }
    i64::from_str_radix(digits, radix).ok()
}

fn parse_iskipspec(spec: &str) -> Option<(i64, i64)> {
    let mut parts = spec.splitn(2, ':');
    let first = parts.next()?;
    let skip1 = expand_number(first)?;
    if let Some(second) = parts.next() {
        let skip2 = expand_number(second)?;
        Some((skip1, skip2))
    } else {
        Some((skip1, skip1))
    }
}

fn diffmsg(
    file1: &str,
    file2: &str,
    byte: i64,
    line: i64,
    b1: u8,
    b2: u8,
    flags: &Flags,
) {
    if flags.sflag {
        // nothing
    } else if flags.bflag {
        println!(
            "{} {} differ: char {}, line {} is {:3o} {:c} {:3o} {:c}",
            file1, file2, byte, line, b1, b1 as char, b2, b2 as char,
        );
    } else {
        println!("{} {} differ: char {}, line {}", file1, file2, byte, line);
    }
}

fn eofmsg(file: &str, flags: &Flags) {
    if !flags.sflag {
        eprintln!("cmp: EOF on {}", file);

fn skip_bytes<R: Read>(mut reader: R, skip: i64, file: &str, flags: &Flags) -> i32 {
    let mut buf = [0u8; 8192];
    let mut remaining = skip;
    while remaining > 0 {
        let to_read = remaining.min(buf.len() as i64) as usize;
        match reader.read(&mut buf[..to_read]) {
            Ok(0) => {
                eofmsg(file, flags);
                return DIFF_EXIT;
            }
            Ok(n) => remaining -= n as i64,
            Err(_) => {
                eprintln!("cmp: {}: {}", file, io::Error::last_os_error());
                return ERR_EXIT;
            }
        }
    }
    OK_EXIT
}
