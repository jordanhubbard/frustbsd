fn getstathz() -> i32 {
    let mut mib: [i32; 2] = [ffi::CTL_KERN, ffi::KERN_CLOCKRATE];
    let mut clockrate: ffi::clockinfo = unsafe { std::mem::zeroed() };
    let mut size: usize = std::mem::size_of::<ffi::clockinfo>();
    if unsafe {
        ffi::sysctl(
            mib.as_ptr(),
            2,
            &mut clockrate as *mut _ as *mut std::os::raw::c_void,
            &mut size,
            std::ptr::null(),
            0,
        )
    } == -1
    {
        eprintln!("time: sysctl kern.clockrate: {}", io::Error::last_os_error());
        exit(1);
    }
    clockrate.stathz
}

fn humantime(out: &mut dyn Write, sec: i64, centisec: i64, decimal_point: char) {
    let mut sec = sec;
    let days = sec / (60 * 60 * 24);
    sec %= 60 * 60 * 24;
    let hrs = sec / (60 * 60);
    sec %= 60 * 60;
    let mins = sec / 60;
    sec %= 60;

    let _ = write!(out, "\t");
    if days != 0 {
        let _ = write!(out, "{}d", days);
    }
    if hrs != 0 {
        let _ = write!(out, "{}h", hrs);
    }
    if mins != 0 {
        let _ = write!(out, "{}m", mins);
    }
    let _ = write!(out, "{}{}{:02}s", sec, decimal_point, centisec);
}

fn showtime(
    out: &mut dyn Write,
    before: &ffi::timespec,
    after: &ffi::timespec,
    ru: &ffi::rusage,
    decimal_point: char,
    pflag: bool,
    hflag: bool,
) {
    let mut after = *after;
    let mut before = *before;

    after.tv_sec -= before.tv_sec;
    after.tv_nsec -= before.tv_nsec;
    if after.tv_nsec < 0 {
        after.tv_sec -= 1;
        after.tv_nsec += 1_000_000_000;
    }

    let real_sec = after.tv_sec;
    let real_cs = after.tv_nsec / 10_000_000;
    let user_sec = ru.ru_utime.tv_sec;
    let user_cs = ru.ru_utime.tv_usec / 10_000;
    let sys_sec = ru.ru_stime.tv_sec;
    let sys_cs = ru.ru_stime.tv_usec / 10_000;

    if pflag {
        let _ = write!(
            out,
            "real {}{}{:02}\nuser {}{}{:02}\nsys {}{}{:02}\n",
            real_sec, decimal_point, real_cs,
            user_sec, decimal_point, user_cs,
            sys_sec, decimal_point, sys_cs,
        );
    } else if hflag {
        humantime(out, real_sec, real_cs, decimal_point);
        let _ = write!(out, " real\t");
        humantime(out, user_sec, user_cs, decimal_point);
        let _ = write!(out, " user\t");
        humantime(out, sys_sec, sys_cs, decimal_point);
        let _ = write!(out, " sys\n");
    } else {
        let _ = write!(
            out,
            "{:>9}{}{:02} real {:>9}{}{:02} user {:>9}{}{:02} sys\n",
            real_sec, decimal_point, real_cs,
            user_sec, decimal_point, user_cs,
            sys_sec, decimal_point, sys_cs,
        );
    }
}
