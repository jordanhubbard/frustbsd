update_user(const char *user, struct timeval secs)
{
	struct user_entry *up, *aup;

	aup = NULL;
	SLIST_FOREACH(up, &Users, next) {
		c = strcmp(up->user, user);
		if (c == 0) {
			timeradd(&up->time, &secs, &up->time);

			return;

	}

}
