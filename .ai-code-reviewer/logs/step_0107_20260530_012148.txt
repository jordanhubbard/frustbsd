use std::process;

fn test_pagesizes() {
    let mut sizes: Vec<usize> = Vec::new();
    // First call to get the number of page sizes
    let nelem = unsafe { libc::getpagesizes(std::ptr::null_mut(), 0) };
    if nelem == -1 {
        eprintln!("getpagesizes(NULL, 0) failed");
        process::exit(1);
    }
    sizes.resize(nelem as usize, 0);
    let nelem = unsafe { libc::getpagesizes(sizes.as_mut_ptr(), nelem) };
    if nelem == -1 {
        eprintln!("getpagesizes failed");
        process::exit(1);
    }
    print!("Supported page sizes:");
    for i in 0..nelem as usize {
        print!(" {}", sizes[i] as i64);
    }
    println!();
}
