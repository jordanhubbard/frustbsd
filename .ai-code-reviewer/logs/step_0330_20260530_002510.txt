//! Rust replacement for stress2 sysctl test case.
//! Calls `sysctl -a` and `fstat` repeatedly to stress the system.

use std::process::Command;
use std::thread;
use std::time::Duration;

fn main() {
    let mut i = 0;
    while i < 64 {
        if let Err(e) = Command::new("sysctl")
            .arg("-a")
            .output()
        {
            eprintln!("sysctl -a: {}: {}:{}", e, file!(), line!());
            std::process::exit(1);
        }

        if let Err(e) = Command::new("fstat")
            .output()
        {
            eprintln!("fstat: {}: {}:{}", e, file!(), line!());
            std::process::exit(1);
        }

        thread::sleep(Duration::from_secs(1));
        i += 1;
    }
}
