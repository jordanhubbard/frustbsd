struct Alloc {
    start: *mut std::ffi::c_void,
    size: usize,
}

impl Alloc {
    fn new() -> Self {
        Alloc { start: ptr::null_mut(), size: 0 }
    }

    fn is_empty(&self) -> bool {
        self.size == 0
    }

    fn realloc_buf(&mut self) {
        let old_size = self.size;
        let new_size = if old_size == 0 {
            ALLOC_FIRST_SIZE
        } else if old_size < PAGE_SIZE {
            2 * old_size
        } else {
            let aligned = (old_size + PAGE_SIZE - 1) & !(PAGE_SIZE - 1);
            aligned + PAGE_SIZE
        };

        let p = unsafe { realloc(self.start, new_size) };
        if p.is_null() {
            eprintln!(
                "cannot realloc allocation (old size: {}, new: {})",
                old_size, new_size
            );
            process::exit(1);
        }
        self.start = p;
        self.size = new_size;
    }

    fn free_buf(&mut self) {
        if !self.is_empty() {
            unsafe { free(self.start) };
            self.start = ptr::null_mut();
            self.size = 0;
        }
    }
}

fn alloc_getpwnam(name: &str, alloc: &mut Alloc) -> Result<libc::passwd, i32> {
    let cname = CString::new(name).unwrap();
    let mut result: *mut libc::passwd = ptr::null_mut();

    if alloc.is_empty() {
        alloc.realloc_buf();
    }

    loop {
        let pwd_ptr = alloc.start as *mut libc::passwd;
        let buf_ptr = unsafe {
            (pwd_ptr as *mut u8).add(std::mem::size_of::<libc::passwd>())
        };
        let buflen = alloc.size - std::mem::size_of::<libc::passwd>();

        let error = unsafe {
            getpwnam_r(
                cname.as_ptr(),
                pwd_ptr,
                buf_ptr as *mut c_char,
                buflen,
                &mut result,
            )
        };

        if error != 0 {
            if error == ERANGE {
                alloc.realloc_buf();
                continue;
            }
            return Err(error);
        }

        if result.is_null() {
            return Err(ENOENT);
        }

        return unsafe { Ok((*result).clone()) };
    }
}

fn parse_user(name: &str) -> uid_t {
    let mut alloc = Alloc::new();
    match alloc_getpwnam(name, &mut alloc) {
        Ok(pwd) => {
            alloc.free_buf();
            return pwd.pw_uid;
        }
        Err(ENOENT) => {}
        Err(e) => {
            alloc.free_buf();
            eprintln!("cannot access the password database: error {}", e);
            process::exit(1);
        }
    }

    match name.parse::<u32>() {
        Ok(uid) => uid,
        Err(_) => {
            eprintln!("invalid UID '{}'", name);
            process::exit(1);
        }
    }
}

fn gid_cmp(a: *const std::ffi::c_void, b: *const std::ffi::c_void) -> c_int {
    let ga = unsafe { *(a as *const gid_t) };
    let gb = unsafe { *(b as *const gid_t) };
    if ga > gb { 1 } else if ga < gb { -1 } else { 0 }
}

fn sort_uniq_groups(groups: &mut Vec<gid_t>) {
    if groups.len() <= 1 {
        return;
    }
    unsafe {
        qsort(
            groups.as_mut_ptr() as *mut std::ffi::c_void,
            groups.len(),
            std::mem::size_of::<gid_t>(),
            gid_cmp,
        );
    }
    let mut j: usize = 0;
    for i in 1..groups.len() {
        if groups[i] != groups[j] {
            j += 1;
            groups[j] = groups[i];
        }
    }
    groups.truncate(j + 1);
}

fn remove_groups(set: &mut Vec<gid_t>, remove: &[gid_t]) {
    if set.is_empty() || remove.is_empty() {
        return;
    }
    let mut from: usize = 0;
    let mut to: usize = 0;
    let mut rem: usize = 0;
    let mut cand = set[0];
    let mut to_rem = remove[0];

    loop {
        if cand < to_rem {
            if to != from {
                set[to] = cand;
            }
            to += 1;
            from += 1;
            if from == set.len() {
                break;
            }
            cand = set[from];
        } else if cand == to_rem {
            from += 1;
            if from == set.len() {
                break;
            }
            rem += 1;
            if rem == remove.len() {
                break;
            }
            cand = set[from];
            to_rem = remove[rem];
        } else {
            rem += 1;
            if rem == remove.len() {
                break;
            }
            to_rem = remove[rem];
        }
    }

    if from == to {
        return;
    }
    let remaining = set.len() - from;
    unsafe {
        memmove(
            set.as_mut_ptr().add(to) as *mut c_char,
            set.as_ptr().add(from) as *const c_char,
            remaining * std::mem::size_of::<gid_t>(),
        );
    }
    set.truncate(to + remaining);
}
