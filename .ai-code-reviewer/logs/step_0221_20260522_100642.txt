use std::fs::File;
use std::io::{self, BufRead, BufReader, Read};
pub fn so_open(n: &str, inc: &[String]) -> Option<BufReader<Box<dyn Read>>> {
    if n == "-" { return Some(BufReader::new(Box::new(io::stdin()))); }
    if let Ok(f) = File::open(n) { return Some(BufReader::new(Box::new(f))); }
    if n.starts_with('/') { eprintln!("soelim: can't open '{}'", n); return None; }
    for d in inc {
        if let Ok(f) = File::open(format!("{}/{}", d, n)) { return Some(BufReader::new(Box::new(f))); }
    }
    eprintln!("soelim: can't open '{}'", n); None
}
