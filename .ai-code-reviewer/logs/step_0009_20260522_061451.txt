use std::ffi::CString;

use std::os::unix::process::CommandExt;
use std::process;

const FILENAME: &str = "nohup.out";
const EXIT_NOEXEC: i32 = 126;
const EXIT_NOTFOUND: i32 = 127;
const EXIT_MISC: i32 = 127;

// Minimal libc constants and FFI to avoid external crate dependency
const STDOUT_FILENO: c_int = 1;
const STDERR_FILENO: c_int = 2;
const O_CREAT: c_int = 0o100;
const O_APPEND: c_int = 0o2000;

type mode_t = u32;
type sighandler_t = c_int;

extern "C" {
    fn isatty(fd: c_int) -> c_int;
    fn dup2(oldfd: c_int, newfd: c_int) -> c_int;
    fn signal(signum: c_int, handler: sighandler_t) -> sighandler_t;

fn usage() -> ! {
    let _ = io::stderr().write_all(b"usage: nohup [--] utility [arguments]\n");
    process::exit(EXIT_MISC);
}

fn dofile() {
    let mut fd: Option<std::fs::File> = None;
    let mut path: String = FILENAME.to_string();

    // Try opening nohup.out in the current directory
    match OpenOptions::new()
        .read(true)
        .write(true)
        .create(true)
        .append(true)
        .open(FILENAME)
    {
        Ok(f) => {
            fd = Some(f);
        }
        Err(_) => {
            // Try $HOME/nohup.out
            if let Some(home) = env::var_os("HOME") {
                if !home.is_empty() {
                    let home_path = home.to_string_lossy().to_string();
                    path = format!("{}/{}", home_path, FILENAME);
                    match OpenOptions::new()
                        .read(true)
                        .write(true)
                        .create(true)
                        .append(true)
                        .open(&path)
                    {
                        Ok(f) => {
                            fd = Some(f);
                        }
                        Err(_) => {
                            eprintln!("can't open a nohup.out file");
                            process::exit(EXIT_MISC);
                        }
                    }
                }
            }
            if fd.is_none() {
                eprintln!("can't open a nohup.out file");
                process::exit(EXIT_MISC);
            }
        }
    }

    let file = fd.unwrap();
    let raw_fd = file.as_raw_fd();

    // dup2(fd, STDOUT_FILENO)
    unsafe {
        dup2(raw_fd, STDOUT_FILENO);
    }

    let _ = io::stderr().write_all(format!("appending output to {}\n", path).as_bytes());
}

fn main() {
    let args: Vec<String> = env::args().collect();

    // Parse arguments: nohup accepts no options except --
    let mut skip = 0;
    if args.len() > 1 && args[1] == "--" {
        skip = 1;
    }

    if args.len() <= skip {
        usage();
    }

    let prog_args: Vec<String> = args[skip..].to_vec();

    // If stdout is a tty, redirect to nohup.out
    if isatty(STDOUT_FILENO) {
        dofile();
    }

    // If stderr is a tty, dup2 stdout to stderr
    if isatty(STDERR_FILENO) {
        unsafe {
            if dup2(STDOUT_FILENO, STDERR_FILENO) == -1 {
                let err = io::Error::last_os_error();
                eprintln!("{}: {}", prog_args[0], err);
                process::exit(EXIT_MISC);
            }
        }
    }

    // Ignore SIGHUP
    unsafe {
        signal(SIGHUP, SIG_IGN);
    }

    // execvp
    let c_args: Vec<CString> = prog_args
        .iter()
        .map(|s| CString::new(s.as_bytes()).unwrap())
        .collect();

    let ret = unsafe {
        execvp(c_args[0].as_ptr(), c_args.as_ptr() as *const *const i8)
    };

    // execvp only returns on error
    let err = io::Error::last_os_error();
    let exit_status = if err.raw_os_error() == Some(ENOENT) {
        EXIT_NOTFOUND
    } else {
        EXIT_NOEXEC
    };
    eprintln!("{}: {}", prog_args[0], err);
    process::exit(exit_status);
