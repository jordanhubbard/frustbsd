struct Flags {

fn expand_number(spec: &str) -> Option<i64> {
    let spec = spec.trim();
    if spec.is_empty() {
        return None;
    }
    // Handle hex prefix
    let (digits, radix) = if spec.starts_with("0x") || spec.starts_with("0X") {
        (&spec[2..], 16)
    } else if spec.starts_with('0') && spec.len() > 1 {
        (spec, 8)
    } else {
        (spec, 10)
    };
    if digits.is_empty() {
        return None;
    }
    i64::from_str_radix(digits, radix).ok()
}

fn parse_iskipspec(spec: &str) -> Option<(i64, i64)> {
    let mut parts = spec.splitn(2, ':');
    let first = parts.next()?;
    let skip1 = expand_number(first)?;
    if let Some(second) = parts.next() {
        let skip2 = expand_number(second)?;
        Some((skip1, skip2))
    } else {
        Some((skip1, skip1))
    }
}

fn diffmsg(
    file1: &str,
    file2: &str,
    byte: i64,
    line: i64,
    b1: u8,
    b2: u8,
    flags: &Flags,
) {
    if flags.sflag {
        // nothing
    } else if flags.bflag {
        println!(
            "{} {} differ: char {}, line {} is {:3o} {:c} {:3o} {:c}",
            file1,
            file2,
            byte,
            line,
            b1,
            b1 as char,
            b2,
            b2 as char,
        );
    } else {
        println!("{} {} differ: char {}, line {}", file1, file2, byte, line);
    }
}

fn eofmsg(file: &str, flags: &Flags) {
    if !flags.sflag {
        eprintln!("cmp: EOF on {}", file);
    }
}

fn compare_regular<R1: Read, R2: Read>(
    mut reader1: BufReader<R1>,
    file1: &str,
    skip1: i64,
    len1: i64,
    mut reader2: BufReader<R2>,
    file2: &str,
    skip2: i64,
    len2: i64,
    limit: i64,
    flags: &Flags,
) -> i32 {
    if skip1 > len1 {
        eofmsg(file1, flags);
        return DIFF_EXIT;
    }
    let len1 = len1 - skip1;
    if skip2 > len2 {
        eofmsg(file2, flags);
        return DIFF_EXIT;
    }
    let len2 = len2 - skip2;

    if flags.sflag && len1 != len2 {
        return DIFF_EXIT;
    }

    // Skip bytes
    let mut buf1 = vec![0u8; 8192];
    let mut buf2 = vec![0u8; 8192];
    let mut remaining_skip1 = skip1;
    let mut remaining_skip2 = skip2;

    loop {
        let to_read = remaining_skip1.min(buf1.len() as i64);
        if to_read <= 0 {
            break;
        }
        match reader1.by_ref().take(to_read as u64).read_to_end(&mut buf1[..to_read as usize]) {
            Ok(n) => {
                if n == 0 {
                    eofmsg(file1, flags);
                    return DIFF_EXIT;
                }
                remaining_skip1 -= n as i64;
            }
            Err(_) => {
                eprintln!("cmp: {}: {}", file1, io::Error::last_os_error());
                return ERR_EXIT;
            }
        }
    }

    loop {
        let to_read = remaining_skip2.min(buf2.len() as i64);
        if to_read <= 0 {
            break;
        }
        match reader2.by_ref().take(to_read as u64).read_to_end(&mut buf2[..to_read as usize]) {
            Ok(n) => {
                if n == 0 {
                    eofmsg(file2, flags);
                    return DIFF_EXIT;
                }
                remaining_skip2 -= n as i64;
            }
            Err(_) => {
                eprintln!("cmp: {}: {}", file2, io::Error::last_os_error());
                return ERR_EXIT;
            }
        }
    }

    let length = len1.min(len2);
    let length = if limit > 0 { length.min(limit) } else { length };

    let mut byte: i64 = 1;
    let mut line: i64 = 1;
    let mut dfound = false;
    let mut buf1 = [0u8; 8192];
    let mut buf2 = [0u8; 8192];
    let mut pos1 = 0usize;
    let mut pos2 = 0usize;
    let mut avail1 = 0usize;
    let mut avail2 = 0usize;
    let mut compared = 0i64;

    while compared < length {
        if pos1 >= avail1 {
            match reader1.read(&mut buf1) {
                Ok(n) => {
                    if n == 0 {
                        eofmsg(file1, flags);
                        return DIFF_EXIT;
                    }
                    avail1 = n;
                    pos1 = 0;
                }
                Err(_) => {
                    eprintln!("cmp: {}: {}", file1, io::Error::last_os_error());
                    return ERR_EXIT;
                }
            }
        }
        if pos2 >= avail2 {
            match reader2.read(&mut buf2) {
                Ok(n) => {
                    if n == 0 {
                        eofmsg(file2, flags);
                        return DIFF_EXIT;
                    }
                    avail2 = n;
                    pos2 = 0;
                }
                Err(_) => {
                    eprintln!("cmp: {}: {}", file2, io::Error::last_os_error());
                    return ERR_EXIT;
                }
            }
        }

        let ch1 = buf1[pos1];
        let ch2 = buf2[pos2];
        pos1 += 1;
        pos2 += 1;
        compared += 1;

        if ch1 != ch2 {
            dfound = true;
            if flags.xflag {
                println!("{:08x} {:02x} {:02x}", byte - 1, ch1, ch2);
            } else if flags.lflag {
                if flags.bflag {
                    println!(
                        "{:6} {:3o} {:c} {:3o} {:c}",
                        byte, ch1, ch1 as char, ch2, ch2 as char
                    );
                } else {
                    println!("{:6} {:3o} {:3o}", byte, ch1, ch2);
                }
            } else {
                diffmsg(file1, file2, byte, line, ch1, ch2, flags);
                return DIFF_EXIT;
            }
        }
        if ch1 == b'\n' {
            line += 1;
        }
        byte += 1;
    }

    if len1 != len2 {
        eofmsg(if len1 > len2 { file2 } else { file1 }, flags);
        return DIFF_EXIT;
    }
    if dfound {
        DIFF_EXIT
    } else {
        OK_EXIT
    }
}

        return DIFF_EXIT;

    } else {

    }
