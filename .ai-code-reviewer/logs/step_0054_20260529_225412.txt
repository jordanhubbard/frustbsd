use std::fs::File;
use std::io::{self, BufRead, BufReader, Read, Write, stdout};
use std::process;

fn usage() -> ! {
    eprintln!("usage: asa [file ...]");
    process::exit(1);
}

fn asa<R: Read>(mut reader: BufReader<R>) {
    let stdout = io::stdout();
    let mut out = stdout.lock();
    let mut buf = Vec::new();
    let mut eol = false;

    loop {
        buf.clear();
        match reader.read_until(b'\n', &mut buf) {
            Ok(0) => break,
            Ok(_) => {}
            Err(e) => {
                eprintln!("asa: {e}");
                process::exit(1);
            }
        }

        if buf.is_empty() {
            continue;
        }

        let first = buf[0];

        // In all cases but '+', terminate previous line, if any
        if first != b'+' && eol {
            let _ = out.write_all(b"\n");
        }

        // Examine and translate the control character
        match first {
            b' ' => {
                // nothing
            }
            b'0' => {
                let _ = out.write_all(b"\n");
            }
            b'1' => {
                let _ = out.write_all(b"\x0c");
            }
            b'+' => {
                // If the '+' is the first character in the input,
                // it shall be equivalent to <space>. (POSIX.1-2017)
                if eol {
                    let _ = out.write_all(b"\r");
                }
            }
            _ => {
                // Treat other characters as <space> per POSIX.1-2017
            }
        }

        // Trim newline if there is one
        let len = buf.len();
        if len > 0 && buf[len - 1] == b'\n' {
            eol = true;
        } else {
            eol = false;
        }

        // Print the rest of the input line (skip the control character)
        if buf.len() > 1 {
            let _ = out.write_all(&buf[1..]);
        }
    }

    // Terminate the last line, if any
    if eol {
        let _ = out.write_all(b"\n");
    }

    // Check for output errors
    if out.flush().is_err() {
        eprintln!("asa: stdout: {}", io::Error::last_os_error());
        process::exit(1);
    }
}

fn main() {
    let args: Vec<String> = std::env::args().collect();

    // asa takes no options
    let files: Vec<String> = args[1..].to_vec();

    let mut exval = 0;

    if files.is_empty() {
        asa(BufReader::new(io::stdin()));
    } else {
        for file in &files {
            if file == "-" {
                asa(BufReader::new(io::stdin()));
            } else {
                match File::open(file) {
                    Ok(f) => asa(BufReader::new(f)),
                    Err(e) => {
                        eprintln!("asa: {file}: {e}");
                        exval = 1;
                    }
                }
            }
        }
    }

    if stdout().flush().is_err() {
        eprintln!("asa: stdout: {}", io::Error::last_os_error());
        process::exit(1);
    }

    process::exit(exval);
}
