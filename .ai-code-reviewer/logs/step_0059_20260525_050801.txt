use std::env;
use std::fs::File;
use std::io::{self, BufRead, BufReader, Read, Write};
use std::process;

#[derive(PartialEq, Eq)]
enum State {
    Init,
    DelimSeen,
    Keyword,
    PuncSeen,
    PuncSeenSvn,
    Text,
}

fn scan<R: Read>(mut reader: BufReader<R>, name: Option<&str>, quiet: bool) -> i32 {
    let mut state = State::Init;
    let mut hasid = false;
    let mut subversion = false;
    let mut buf = String::new();

    let stdin_name = if name.is_none() {
        "standard input"
    } else {
        ""
    };

    if let Some(n) = name {
        print!("{}:\n", n);
        if io::stdout().flush().is_err() {
            eprintln!("{}: stdout: {}", getprogname(), io::Error::last_os_error());
            return 1;
        }
    }

    let bytes = reader.bytes();
    for result in bytes {
        let c = match result {
            Ok(b) => b as u8,
            Err(_) => break,
        };

        match state {
            State::Init => {
                if c == b'$' {
                    state = State::DelimSeen;
                }
                // Otherwise stay in Init
            }
            State::DelimSeen => {
                if c.is_ascii_alphabetic() {
                    buf.clear();
                    buf.push('$' as char);
                    buf.push(c as char);
                    state = State::Keyword;
                } else if c == b'$' {
                    // Stay in DelimSeen
                } else {
                    state = State::Init;
                }
            }
            State::Keyword => {
                buf.push(c as char);
                if c.is_ascii_alphabetic() {
                    // Stay in Keyword
                } else if c == b':' {
                    state = State::PuncSeen;
                    subversion = false;
                } else if c == b'$' {
                    state = State::DelimSeen;
                } else {
                    state = State::Init;
                }
            }
            State::PuncSeen | State::PuncSeenSvn => {
                buf.push(c as char);
                match c {
                    b':' => {
                        if state == State::PuncSeen {
                            state = State::PuncSeenSvn;
                            subversion = true;
                        } else {
                            state = State::Init;
                        }
                    }
                    b' ' => {
                        state = State::Text;
                    }
                    _ => {
                        state = State::Init;
                    }
                }
            }
            State::Text => {
                buf.push(c as char);
                if c.is_ascii_control() {
                    state = State::Init;
                } else if c == b'$' {
                    // Valid ident should end with a space.
                    // Subversion extension uses '#' to indicate overflow.
                    let last_char = buf.as_bytes().get(buf.len() - 2).copied();
                    if last_char == Some(b' ') || (subversion && last_char == Some(b'#')) {
                        print!("     {}\n", buf);
                        if io::stdout().flush().is_err() {
                            eprintln!("{}: stdout: {}", getprogname(), io::Error::last_os_error());
                            return 1;
                        }
                        hasid = true;
                    }
                    state = State::Init;
                }
                // Other characters: stay in Text
            }
        }
    }

    if !hasid {
        if !quiet {
            let display_name = name.unwrap_or(stdin_name);
            eprintln!(
                "{} warning: no id keywords in {}",
                getprogname(),
                display_name
            );
        }
        return 1;
    }

    0
}

fn getprogname() -> String {
    env::args()
        .next()
        .unwrap_or_else(|| "ident".to_string())
}

fn usage() {
    eprintln!("usage: {} [-q] [-V] [file...]", getprogname());
    process::exit(1);
}

fn main() {
    let args: Vec<String> = env::args().collect();
    let progname = &args[0];
    let mut quiet = false;
    let mut files: Vec<String> = Vec::new();

    let mut i = 1;
    while i < args.len() {
        let arg = &args[i];
        if arg == "-q" || arg == "--quiet" {
            quiet = true;
        } else if arg == "-V" || arg == "--version" {
            // Compat with GNU rcs's ident
            process::exit(0);
        } else if arg.starts_with('-') {
            usage();
        } else {
            files.push(arg.clone());
        }
        i += 1;
    }

    let ret = if files.is_empty() {
        let stdin = io::stdin();
        let reader = BufReader::new(stdin.lock());
        scan(reader, None, quiet)
    } else {
        let mut ret = 0;
        for fname in &files {
            match File::open(fname) {
                Ok(f) => {
                    let reader = BufReader::new(f);
                    if scan(reader, Some(fname), quiet) != 0 {
                        ret = 1;
                    }
                }
                Err(e) => {
                    eprintln!("{}: {}: {}", progname, fname, e);
                    process::exit(1);
                }
            }
        }
        ret
    };

    process::exit(ret);
}
