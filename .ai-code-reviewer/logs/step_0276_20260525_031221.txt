use std::fs::{self, File, OpenOptions};

fn dec(c: u8) -> u8 {
    if c == b'`' {
        0
    } else {
        c - b' '
    }
}

fn base64_decode_char(c: u8) -> u8 {
    match c {
        b'A'..=b'Z' => c - b'A',
        b'a'..=b'z' => c - b'a' + 26,
        b'0'..=b'9' => c - b'0' + 52,
        b'+' => 62,
        b'/' => 63,
        _ => 0,
    }
}

fn decode_body(input: &mut dyn BufRead, output: &mut dyn Write, mode: u16, remotefile: &str, raw: bool) {
    if !raw {
        // Skip the "begin ..." header line
        let mut header = String::new();
        if input.read_line(&mut header).is_err() {
            return;
        }
    }

    let mut buf = [0u8; 45];
    loop {
        let mut line = String::new();
        match input.read_line(&mut line) {
            Ok(0) => break,
            Ok(_) => {
                let line_bytes = line.as_bytes();
                if line_bytes.len() < 2 {
                    break;
                }
                let n = dec(line_bytes[0]) as usize;
                if n == 0 {
                    break;
                }
                let mut i = 1;
                let mut out_idx = 0;
                while i + 3 < line_bytes.len() && out_idx < n {
                    let b0 = dec(line_bytes[i]);
                    let b1 = dec(line_bytes[i + 1]);
                    let b2 = dec(line_bytes[i + 2]);
                    let b3 = dec(line_bytes[i + 3]);

                    buf[out_idx] = (b0 << 2) | (b1 >> 4);
                    out_idx += 1;
                    if out_idx < n {
                        buf[out_idx] = ((b1 & 0x0F) << 4) | (b2 >> 2);
                        out_idx += 1;
                    }
                    if out_idx < n {
                        buf[out_idx] = ((b2 & 0x03) << 6) | b3;
                        out_idx += 1;
                    }
                    i += 4;
                }
                output.write_all(&buf[..out_idx]).unwrap();
            }
            Err(_) => break,
        }
    }

    if !raw {
        // Skip "end" line
        let mut endline = String::new();
        let _ = input.read_line(&mut endline);
    }

    // Set permissions on output file
    if let Some(path) = remotefile {
        if let Ok(p) = Path::new(path).canonicalize() {
            let _ = fs::set_permissions(&p, fs::Permissions::from_mode(mode));
        }
    }
}

fn base64_decode_body(input: &mut dyn BufRead, output: &mut dyn Write, mode: u16, remotefile: &str, raw: bool) {
    if !raw {
        let mut header = String::new();
        if input.read_line(&mut header).is_err() {
            return;
        }
    }

    let mut buf = [0u8; 3];
    let mut carry = 0usize;
    let mut carry_buf = [0u8; 3];

    loop {
        let mut line = String::new();
        match input.read_line(&mut line) {
            Ok(0) => break,
            Ok(_) => {
                let line_bytes = line.trim_end().as_bytes();
                if line_bytes.is_empty() {
                    break;
                }
                let mut i = 0;
                while i < line_bytes.len() {
                    let c0 = if i < line_bytes.len() { line_bytes[i] } else { b'=' };
                    let c1 = if i + 1 < line_bytes.len() { line_bytes[i + 1] } else { b'=' };
                    let c2 = if i + 2 < line_bytes.len() { line_bytes[i + 2] } else { b'=' };
                    let c3 = if i + 3 < line_bytes.len() { line_bytes[i + 3] } else { b'=' };

                    let v0 = base64_decode_char(c0);
                    let v1 = base64_decode_char(c1);
                    let v2 = base64_decode_char(c2);
                    let v3 = base64_decode_char(c3);

                    let eq2 = c2 == b'=';
                    let eq3 = c3 == b'=';

                    carry_buf[carry] = (v0 << 2) | (v1 >> 4);
                    carry += 1;
                    if !eq2 {
                        carry_buf[carry] = ((v1 & 0x0F) << 4) | (v2 >> 2);
                        carry += 1;
                    }
                    if !eq3 {
                        carry_buf[carry] = ((v2 & 0x03) << 6) | v3;
                        carry += 1;
                    }

                    if carry == 3 {
                        output.write_all(&carry_buf).unwrap();
                        carry = 0;
                    }
                    i += 4;
                }
            }
            Err(_) => break,
        }
    }

    if carry > 0 {
        output.write_all(&carry_buf[..carry]).unwrap();
    }

    if !raw {
        let mut endline = String::new();
        let _ = input.read_line(&mut endline);
    }

    }
