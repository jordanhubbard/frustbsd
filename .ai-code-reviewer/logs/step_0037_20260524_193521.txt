use std::env;
use std::ffi::CString;
use std::os::raw::c_char;
use std::process;

extern "C" {
    fn basename(s: *const c_char) -> *mut c_char;
}

fn stripsuffix(p: &mut String, suffix: &str) {
    if suffix.is_empty() {
        return;
    }
    let plen = p.len();
    let slen = suffix.len();
    if slen >= plen {
        return;
    }
    let start = plen - slen;
    // Check that suffix matches at the end
    if &p[start..] != suffix {
        return;
    }
    // Ensure the match occurred on a character boundary
    if !p.is_char_boundary(start) {
        return;
    }
    p.truncate(start);
}

fn usage() {
    eprintln!(
        "usage: basename string [suffix]\n\
         \tbasename [-a] [-s suffix] string [...]"
    );
    process::exit(1);
}

fn main() {
    let args: Vec<String> = env::args().collect();
    let mut aflag = false;
    let mut suffix: Option<String> = None;
    let mut optind = 1;

    let mut i = 1;
    while i < args.len() {
        let arg = &args[i];
        if arg == "-a" {
            aflag = true;
            i += 1;
        } else if arg.starts_with("-s") {
            if arg.len() == 2 {
                // -s suffix (separate argument)
                if i + 1 >= args.len() {
                    usage();
                }
                suffix = Some(args[i + 1].clone());
                i += 2;
            } else {
                // -ssuffix (combined)
                suffix = Some(arg[2..].to_string());
                i += 1;
            }
        } else if arg.starts_with('-') {
            usage();
        } else {
            optind = i;
            break;
        }
    }

    let remaining: Vec<String> = args[optind..].to_vec();

    if remaining.is_empty() {
        usage();
    }

    if remaining[0].is_empty() {
        println!();
        process::exit(0);
    }

    // Handle POSIX two-arg form: basename string [suffix]
    let suffix = if suffix.is_none() && !aflag && remaining.len() == 2 {
        Some(remaining[1].clone())
    } else {
        suffix
    };

    let suffix_str = suffix.as_deref().unwrap_or("");

    // Determine which arguments are strings to process.
    // In POSIX two-arg form (no -a, no -s, exactly 2 args), the second arg is suffix.
