use std::ffi::CString;
use std::io::{self, Write};
use std::os::raw::{c_char, c_int, mode_t};
use std::path::Path;
use std::process;

extern "C" {
    fn mkdir(path: *const c_char, mode: mode_t) -> c_int;
    fn chmod(path: *const c_char, mode: mode_t) -> c_int;
    fn umask(mask: mode_t) -> mode_t;
    fn stat(path: *const c_char, buf: *mut stat_buf) -> c_int;
}

#[repr(C)]
struct stat_buf {
    st_mode: mode_t,
    // We only need st_mode, but stat has more fields
    _padding: [u8; 256],
}

const S_IRWXU: mode_t = 0o700;
const S_IRWXG: mode_t = 0o070;
const S_IRWXO: mode_t = 0o007;
const S_IWUSR: mode_t = 0o200;
const S_IXUSR: mode_t = 0o100;
const S_ISDIR: mode_t = 0o040000;

fn usage() -> ! {
    let msg = "usage: mkdir [-pv] [-m mode] directory_name ...\n";
    let _ = io::stderr().write_all(msg.as_bytes());
    process::exit(64); // EX_USAGE
}

fn parse_mode(mode_str: &str) -> Option<mode_t> {
    let mode_str = mode_str.trim_start_matches('0');
    if mode_str.is_empty() {
        return Some(0);
    }
    // Parse octal mode
    let mut result: mode_t = 0;
    for ch in mode_str.chars() {
        if !('0'..='7').contains(&ch) {
            return None;
        }
        result = (result << 3) | (ch as mode_t - '0' as mode_t);
    }
    Some(result)
}

fn warn(msg: &str) {
    let _ = writeln!(io::stderr(), "mkdir: {}", msg);
}

fn build(path: &str, omode: mode_t, verbose: bool) -> i32 {
    let mut p = path.to_string();
    let mut oumask: mode_t = 0;
    let mut retval: i32 = 1;
    let mut first = true;

    // Skip leading '/'
    let start = if p.starts_with('/') { 1 } else { 0 };

    let bytes = p.as_bytes();
    let mut i = start;

    while i < bytes.len() {
        let is_last = i == bytes.len() - 1;
        let is_separator = bytes[i] == b'/';

        if !is_last && !is_separator {
            i += 1;
            continue;
        }

        if is_last || (is_separator && i + 1 >= bytes.len()) {
            // Process this segment
            let seg_end = if is_last { bytes.len() } else { i };
            let seg = &p[..seg_end];

            if first {
                oumask = unsafe { umask(0) };
                let numask = oumask & !(S_IWUSR | S_IXUSR);
                unsafe { umask(numask) };
                first = false;
            }

            if is_last {
                unsafe { umask(oumask) };
            }

            let c_path = CString::new(seg).unwrap();
            let mode = if is_last { omode } else { S_IRWXU | S_IRWXG | S_IRWXO };

            let ret = unsafe { mkdir(c_path.as_ptr(), mode) };
            if ret < 0 {
                // Check errno
                let err = io::Error::last_os_error();
                let raw = err.raw_os_error().unwrap_or(0);
                if raw == 17 || raw == 21 { // EEXIST or EISDIR
                    // Check if it's a directory
                    let mut sb = stat_buf { st_mode: 0, _padding: [0; 256] };
                    if unsafe { stat(c_path.as_ptr(), &mut sb) } < 0 {
                        warn(seg);
                        retval = 0;
                        break;
                    } else if (sb.st_mode & S_ISDIR) == 0 {
                        if is_last {
                            let _ = writeln!(io::stderr(), "mkdir: {}: File exists", seg);
                        } else {
                            let _ = writeln!(io::stderr(), "mkdir: {}: Not a directory", seg);
                        }
                        retval = 0;
                        break;
                    }
                    if is_last {
                        retval = 2;
                    }
                } else {
                    warn(seg);
                    retval = 0;
                    break;
                }
            } else if verbose {
                let _ = writeln!(io::stdout(), "{}", seg);
            }

            if !is_last {
                // Restore the '/'
                p.as_bytes_mut()[i] = b'/';
            }
        }

        i += 1;
    }

    if !first && retval != 0 {
        unsafe { umask(oumask) };
    }

    retval
}

fn main() {
    let args: Vec<String> = std::env::args().skip(1).collect();

    let mut mode_str: Option<&str> = None;
    let mut pflag = false;
    let mut verbose = false;

    let mut i = 0;
    let mut dirs: Vec<String> = Vec::new();

    while i < args.len() {
        match args[i].as_str() {
            "-m" => {
                i += 1;
                if i >= args.len() {
                    usage();
                }
                mode_str = Some(&args[i]);
            }
            "-p" => pflag = true,
            "-v" => verbose = true,
            s if s.starts_with('-') => usage(),
            s => dirs.push(s.to_string()),
        }
        i += 1;
    }

    if dirs.is_empty() {
        usage();
    }

    let omode = match mode_str {
        Some(m) => match parse_mode(m) {
            Some(mode) => mode,
            None => {
                eprintln!("mkdir: invalid file mode: {}", m);
                process::exit(1);
            }
        },
        None => S_IRWXU | S_IRWXG | S_IRWXO,
    };

    let mut exitval = 0;

    for dir in &dirs {
        let success = if pflag {
            let ret = build(dir, omode, verbose);
            ret != 0
        } else {
            let c_path = match CString::new(dir.as_bytes()) {
                Ok(p) => p,
                Err(_) => {
                    warn(dir);
                    exitval = 1;
                    continue;
                }
            };

            let ret = unsafe { mkdir(c_path.as_ptr(), omode) };
            if ret < 0 {
                let err = io::Error::last_os_error();
                let raw = err.raw_os_error().unwrap_or(0);
                if raw == 2 || raw == 20 { // ENOENT or ENOTDIR
                    // Try to warn about parent
                    if let Some(parent) = Path::new(dir).parent() {
                        warn(parent.to_string_lossy().as_ref());
                    } else {
                        warn(dir);
                    }
                } else {
                    warn(dir);
                }
                false
            } else {
                if verbose {
                    let _ = writeln!(io::stdout(), "{}", dir);
                }
                true
            }
        };

        if !success {
            exitval = 1;
        } else if mode_str.is_some() {
            // Apply chmod for special bits
            let c_path = CString::new(dir.as_bytes()).unwrap();
            if unsafe { chmod(c_path.as_ptr(), omode) } < 0 {
                warn(dir);
                exitval = 1;
            }
        }
    }

    process::exit(exitval);
}
