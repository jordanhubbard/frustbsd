#[derive(Clone, Copy, PartialEq)]

struct Opts {

fn usage() -> ! {
    eprintln!("usage: uniq [-cdiu] [-D[septype]] [-f fields] [-s chars] [input [output]]");
    process::exit(1);
}

fn strtonum(s: &str, low: i64, high: i64) -> Result<i64, String> {
    match s.parse::<i64>() {
        Ok(n) if n >= low && n <= high => Ok(n),
        Ok(_) => Err(format!("out of range for {}..{}", low, high)),
        Err(_) => Err("invalid number".to_string()),
    }
}

fn make_key(line: &str, opts: &Opts) -> String {
    let mut s = line.trim_end_matches('\n').to_string();
    if opts.numfields > 0 || opts.numchars > 0 {
        let mut chars = s.chars();
        let mut fs = 0;
        while fs < opts.numfields {
            while let Some(c) = chars.clone().next() {
                if c == ' ' || c == '\t' { chars.next(); } else { break; }
            }
            let mut got = false;
            while let Some(c) = chars.clone().next() {
                if c == ' ' || c == '\t' { break; }
                chars.next(); got = true;
            }
            if !got { break; }
            fs += 1;
        }
        for _ in 0..opts.numchars { chars.next(); }
        s = chars.collect();
    }
    if opts.iflag { s = s.to_lowercase(); }
    s
}

fn obsolete_args(args: &[String]) -> Vec<String> {
    let mut r = Vec::new();
    for a in args {
        if a == "--" || (!a.starts_with('-') && !a.starts_with('+')) {
            r.push(a.clone()); continue;
        }
        if a.starts_with("--") { r.push(a.clone()); continue; }
        let rest = &a[1..];
        if rest.chars().next().map_or(false, |c| c.is_ascii_digit()) {
            let flag = if a.starts_with('+') { 's' } else { 'f' };
            r.push(format!("-{}{}", flag, rest));
        } else { r.push(a.clone()); }
    }
    r
}
