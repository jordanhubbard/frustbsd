fn clock_gettime(id: c_int) -> Result<ffi::timespec, std::io::Error> {
    let mut ts: ffi::timespec = unsafe { std::mem::zeroed() };
    let ret = unsafe { ffi::clock_gettime(id, &mut ts) };
    if ret != 0 {
        Err(std::io::Error::last_os_error())
    } else {
        Ok(ts)
    }
}

fn getstathz() -> Result<c_int, std::io::Error> {
    let mib = [ffi::CTL_KERN, ffi::KERN_CLOCKRATE];
    let mut clockrate: ffi::clockinfo = unsafe { std::mem::zeroed() };
    let mut size: usize = std::mem::size_of::<ffi::clockinfo>();
    let ret = unsafe {
        ffi::sysctl(
            mib.as_ptr(),
            2,
            &mut clockrate as *mut _ as *mut c_char,
            &mut size,
            std::ptr::null(),
            0,
        )

        Err(std::io::Error::last_os_error())
    } else {
        Ok(clockrate.stathz)
    }
}

fn humantime(out: &mut dyn Write, sec: c_long, centisec: c_long) {
    let dp = unsafe { DECIMAL_POINT as char };
    let mut s = sec;
    let days = s / (60 * 60 * 24);
    s %= 60 * 60 * 24;
    let hrs = s / (60 * 60);
    s %= 60 * 60;
    let mins = s / 60;
    s %= 60;

    if days != 0 { write!(out, "{}d", days).ok(); }
    if hrs != 0 { write!(out, "{}h", hrs).ok(); }
    if mins != 0 { write!(out, "{}m", mins).ok(); }

}

    ru: &ffi::rusage, hflag: bool, pflag: bool) {

    elapsed.tv_sec -= before.tv_sec;
    elapsed.tv_nsec -= before.tv_nsec;

        elapsed.tv_nsec += 1_000_000_000;

        writeln!(out, "sys {}{}{:02}", ru.ru_stime.tv_sec, dp, ru.ru_stime.tv_usec / 10_000).ok();
    } else if hflag {
        humantime(out, elapsed.tv_sec, elapsed.tv_nsec / 10_000_000);
        write!(out, " real\t").ok();

        write!(out, " user\t").ok();

        writeln!(out, " sys").ok();
