//! Rust replacement for FreeBSD_version example program.
//!
//! Prints the FreeBSD version at compile time and runtime.

extern "C" {
    fn getosreldate() -> u32;
}

fn main() {
    // Compile-time FreeBSD version
    // In Rust, we approximate this with the version from the build environment
    // or use a default that matches the C behavior
    let compile_version = option_env!("OSRELDATE")
        .and_then(|v| v.parse::<u32>().ok())
        .unwrap_or_else(|| unsafe { getosreldate() });

    println!("Compilation release date: {}", compile_version);

    // Runtime FreeBSD version
    let runtime_version = unsafe { getosreldate() };
    println!("Execution environment release date: {}", runtime_version);
}
