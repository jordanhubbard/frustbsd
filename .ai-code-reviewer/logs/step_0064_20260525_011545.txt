            "-k" => host = Some(KNICHOST),
            "-l" => host = Some(LNICHOST),
            "-m" => host = Some(MNICHOST),
            "-p" => { i += 1; if i >= args.len() { usage(); } port = &args[i]; }
            "-P" => host = Some(PDBHOST),
            "-Q" => flags |= WHOIS_QUICK,
            "-r" => host = Some(RNICHOST),
            "-R" => flags |= WHOIS_RECURSE,

        i += 1;
    }

    if host.is_none() && country.is_none() {

            if (flags & WHOIS_QUICK) == 0 { flags |= WHOIS_RECURSE; }
        }

            whois(query, &qnichost, port, flags, &mut visited);
        } else {

                Some(h) => h,
                None => choose_server(query),
