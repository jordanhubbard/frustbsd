static void
hostconv(char *arg)
{
	static int first = 1;
	static char *hostdot, name[MAXHOSTNAMELEN];
	char *argdot;

	if ((argdot = strchr(arg, '.')) == NULL)
		return;
	if (first) {
		first = 0;
		if (gethostname(name, sizeof(name)))
			xo_err(1, "gethostname");
		hostdot = strchr(name, '.');
	}
	if (hostdot != NULL && strcasecmp(hostdot, argdot) == 0)
		*argdot = '\0';
}
