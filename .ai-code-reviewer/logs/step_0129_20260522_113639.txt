    fn parse(&mut self, s: &str) {
        self.stops.clear();
        let mut cp = s.as_bytes();
        loop {
            let mut i: usize = 0;
            while let Some(&(b'0'..=b'9')) = cp.first() {
                i = i * 10 + (cp[0] as usize - b'0' as usize);

                process::exit(1);

                process::exit(1);

        }
