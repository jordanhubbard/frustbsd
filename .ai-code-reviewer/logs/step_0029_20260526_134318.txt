use std::ffi::CString;
use std::os::raw::c_int;
use std::process;

extern "C" {
    fn modfind(name: *const i8) -> c_int;
    fn modstat(modid: c_int, msp: *mut module_stat) -> c_int;
    fn syscall(number: c_int, ...) -> c_int;
}

// MAXPATHLEN on FreeBSD is 1024
const MAXPATHLEN: usize = 1024;

#[repr(C)]
union modspecific_t {
    intval: c_int,
    uintval: u32,
    longval: std::os::raw::c_long,
    ulongval: std::os::raw::c_ulong,
}

#[repr(C)]
struct module_stat {
    version: c_int,
    name: [std::os::raw::c_char; MAXPATHLEN],
    refs: c_int,
    id: c_int,
    data: modspecific_t,
}

fn main() {
    let name = match CString::new("sys/syscall") {
        Ok(n) => n,
        Err(_) => {
            eprintln!("modfind: invalid name");
            process::exit(1);
        }
    };

    let modid = unsafe { modfind(name.as_ptr()) };
    if modid == -1 {
        eprintln!("modfind: failed");
        process::exit(1);
    }

    let mut stat: module_stat = unsafe { std::mem::zeroed() };
    stat.version = std::mem::size_of::<module_stat>() as c_int;

    if unsafe { modstat(modid, &mut stat) } != 0 {
        eprintln!("modstat: failed");
        process::exit(1);
    }

    let syscall_num = unsafe { stat.data.intval };
    let result = unsafe { syscall(syscall_num) };
    process::exit(result as i32);
}
