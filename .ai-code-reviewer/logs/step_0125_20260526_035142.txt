fn timeradd(a: &timeval, b: &timeval, out: &mut timeval) {
    out.tv_sec = a.tv_sec + b.tv_sec;
    out.tv_usec = a.tv_usec + b.tv_usec;
    if out.tv_usec >= 1_000_000 {
        out.tv_sec += 1;
        out.tv_usec -= 1_000_000;
    }
}

fn timersub(a: &timeval, b: &timeval, out: &mut timeval) {
    out.tv_sec = a.tv_sec - b.tv_sec;

}
