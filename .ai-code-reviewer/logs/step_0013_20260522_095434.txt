use std::env;
use std::fs::File;
use std::io::{self, BufRead, BufReader, Read, Write, Stdin};
use std::path::Path;

const C_OPTION: u8 = 0x1;

fn usage() -> ! {
    writeln!(io::stderr(), "usage: soelim [-Crtv] [-I dir] [files]").unwrap();
    std::process::exit(1);
}

fn soelim_fopen(name: &str, includes: &[String]) -> Result<BufReader<File>, ()> {
    if name == "-" {
        // We handle stdin specially by returning an error that the caller
        // interprets as "use stdin".
        return Err(());
    }

    // Try opening the file directly
    if let Ok(f) = File::open(name) {
        return Ok(BufReader::new(f));
    }

    // If absolute path, just warn and fail
    if name.starts_with('/') {
        eprintln!("soelim: can't open '{}': No such file or directory", name);
        return Err(());
    }

    // Try include directories
    for inc_dir in includes {
        let path = format!("{}/{}", inc_dir, name);
        if let Ok(f) = File::open(&path) {
            return Ok(BufReader::new(f));
        }
    }

    eprintln!("soelim: can't open '{}': No such file or directory", name);
    Err(())
}

fn soelim_file<R: Read>(
    reader: Option<BufReader<R>>,
    stdin: &mut Option<BufReader<Stdin>>,
    includes: &[String],
    flags: u8,
) -> bool
where
    R: Read,
{
    let mut reader = match reader {
        Some(r) => r,
        None => match stdin.take() {
            Some(s) => s,
            None => return true,
        },
    };

    let mut line = String::new();
    loop {
        line.clear();
        match reader.read_line(&mut line) {
            Ok(0) => break,
            Ok(_) => {}
            Err(_) => return true,
        }

        // Check for .so directive
        if !line.starts_with(".so") {
            print!("{}", line);
            continue;
        }

        let walk = &line[3..];
        if !walk.chars().next().map_or(false, |c| c.is_whitespace()) && (flags & C_OPTION) == 0 {
            print!("{}", line);
            continue;
        }

        // Skip whitespace
        let walk = walk.trim_start();

        // Extract the filename (up to next whitespace or end)
        let mut end = 0;
        for (i, c) in walk.char_indices() {
            if c.is_whitespace() {
                end = i;
                break;
            }
        }
        let filename = &walk[..end];
        let rest = if end < walk.len() { &walk[end..] } else { "" };

        if filename.is_empty() {
            print!("{}", line);
            continue;
        }

        // Try to open the included file
        if filename == "-" {
            // Recurse with stdin - but stdin is already consumed
            // The original C code would reopen stdin, but in practice
            // this is unusual. We handle it by recursing with stdin.
            if soelim_file(None, stdin, includes, flags) {
                return true;
            }
        } else {
            match soelim_fopen(filename, includes) {
                Ok(f) => {
                    if soelim_file(Some(f), &mut None, includes, flags) {
                        return true;
                    }
                }
                Err(_) => return true,
            }
        }

        // Print the rest of the line after the filename
        if !rest.is_empty() {
            print!("{}", rest);
        }
    }

    false
}

    let args: Vec<String> = env::args().collect();
    let mut flags: u8 = 0;
    let mut includes: Vec<String> = Vec::new();

    let mut i = 1;
    let mut file_args: Vec<String> = Vec::new();

    while i < args.len() {
        let arg = &args[i];
        if arg.starts_with('-') && arg.len() > 1 {
            // Check if it's a long option or combined short options
            if arg == "-" {
                // Explicit stdin
                file_args.push("-".to_string());
                i += 1;
                continue;
            }

            let mut chars = arg[1..].chars();
            let first = match chars.next() {
                Some(c) => c,
                None => usage(),

                'C' => flags |= C_OPTION,
                'r' | 'v' | 't' => {

                    // Check if there's a value after the flag
                    let remaining: String = chars.collect();
                    if !remaining.is_empty() {
                        includes.push(remaining);
                    } else {
                        i += 1;
                        if i >= args.len() {
                            usage();
                        }
                        includes.push(args[i].clone());
                    }
                }
                _ => usage(),
            }
        } else {
            file_args.push(arg.clone());
        }
        i += 1;
    }

    let mut ret = 0;

    if file_args.is_empty() {
        // Read from stdin
        let stdin = io::stdin();
        let reader = BufReader::new(stdin);
        if soelim_file(Some(reader), &mut None, &includes, flags) {
            ret = 1;

                            ret = 1;
                        }
                    }
                    Err(_) => {
                        ret = 1;
                    }
                }
            }
        }
    }

    std::process::exit(ret);
