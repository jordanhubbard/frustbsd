fn gettabs(arg: &str, stops: &mut Vec<i64>) {
    let mut last: i64 = 0;
    let mut nstops: usize = 0;
    let mut first = true;

    for tok in arg.split(',') {
        if nstops >= NSTOPS {
            errx(1, &format!("too many tab stops (limit {})", NSTOPS));
        }
        let tok_trimmed = tok.trim();
        if tok_trimmed.is_empty() {
            continue;
        }
        let stop: i64 = match tok_trimmed.parse() {
            Ok(v) => v,
            Err(_) => errx(1, &format!("{}: invalid tab stop", tok_trimmed)),
        };
        if stop <= 0 {
            errx(1, &format!("{}: invalid tab stop", tok_trimmed));
        }
        let final_stop = if tok_trimmed.starts_with('+') {
            if first {
                errx(1, &format!("{}: first tab may not be relative", tok_trimmed));
            }
            stop + last
        } else {
            stop
        };
        if last > final_stop {
            errx(1, "cannot go backwards");
        }
        last = final_stop;
        stops.push(final_stop);
        nstops += 1;
        first = false;
    }
}

fn ttywidth() -> i64 {
    let stdout = io::stdout();
    let fd = stdout.as_raw_fd();
    let mut ws: Winsize = unsafe { std::mem::zeroed() };
    if unsafe { ioctl(fd, TIOCGWINSZ, &mut ws as *mut Winsize) } != -1 {
        return ws.ws_col as i64;
    }
    let width = unsafe { tgetnum(c"co") } as i64;
    if width != 0 {
        return width;
    }
    warnx("cannot find terminal width; defaulted to 80");
    80
}
