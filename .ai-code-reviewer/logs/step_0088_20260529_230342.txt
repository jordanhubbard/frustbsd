fn cstr_to_bytes(s: *const c_char, len: usize) -> [u8; 16] {
    let mut buf = [0u8; 16];
    if !s.is_null() {
        let cstr = unsafe { CStr::from_ptr(s) };
        let bytes = cstr.to_bytes();
        let copy_len = bytes.len().min(len);
        buf[..copy_len].copy_from_slice(&bytes[..copy_len]);
    }
    buf
}

fn entry_from_utmpx(ut: *const utmpx) -> LastLoginEntry {
    unsafe {
        LastLoginEntry {
            user: cstr_to_bytes((*ut).ut_user.as_ptr(), 16),
            line: cstr_to_bytes80((*ut).ut_line.as_ptr()),
            host: cstr_to_bytes128((*ut).ut_host.as_ptr()),
            tv_sec: (*ut).ut_tv.tv_sec,
        }
    }
}

fn bytes_to_str(buf: &[u8]) -> &str {
    let end = buf.iter().position(|&b| b == 0).unwrap_or(buf.len());
    std::str::from_utf8(&buf[..end]).unwrap_or("")
}

fn output(entry: &LastLoginEntry) {
    let stdout = io::stdout();
    let mut out = stdout.lock();
    let user = bytes_to_str(&entry.user);
    let line = bytes_to_str(&entry.line);
    let host = bytes_to_str(&entry.host);
    let t = entry.tv_sec;

    let ctime_str = unsafe {
        let ptr = ctime(&t as *const time_t);
        if ptr.is_null() {
            String::from("(unknown)")
        } else {
            let cstr = CStr::from_ptr(ptr);
            cstr.to_string_lossy().into_owned()
        }
    };

    let _ = writeln!(
        out,
        "{:<10} {:<8} {:<22.22} {}",
        user, line, host, ctime_str
    );
    if out.flush().is_err() {
        eprintln!("lastlogin: stdout: {}", io::Error::last_os_error());
        process::exit(1);
    }
}
