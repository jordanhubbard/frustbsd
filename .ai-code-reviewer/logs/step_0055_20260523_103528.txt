fn putc(ch: char) {
    let stdout = io::stdout();
    let mut handle = stdout.lock();
    if handle.write_all(ch.encode_utf8(&mut [0u8; 4]).as_bytes()).is_err() {
        eprintln!("write error");
        process::exit(1);
    }
}

fn addto_lineno(lno: &mut i32, offset: i32) {
    if offset > 0 {
        if *lno >= i32::MAX - offset {
            eprintln!("too many lines");
            process::exit(1);
        }
    } else {
        if *lno < i32::MIN - offset {
            eprintln!("too many reverse line feeds");
            process::exit(1);
        }
    }
    *lno += offset;
}

fn flush_line(l: &Line) {

        let mut sorted: Vec<CharInfo> = l.l_line.clone();

        c = &sorted;
        nchars = sorted.len();

        nchars = l.l_line_len;

    let mut idx: usize = 0;

        let this_col = c[idx].c_column;

            endc += 1;

                idx = endc;
                continue;
            }

                        break;

                    if tab_size == 1 {
                        putc(SPACE);
                    } else {
                        putc(TAB);
                    }
                    nspace -= tab_size;
                    last_col = tab_col;
                }
            }
            for _ in 0..nspace {
                putc(SPACE);
            }
            last_col = this_col;
        }

                    CSet::Normal => putc(SI),
                    CSet::Alternate => putc(SO),

                    putc(BS);
                }
            }
            ci += 1;
        }

        idx = endc;
    }

fn flush_lines(nflush: &mut i32) {
    let mut remaining = unsafe { std::ptr::read(&LINES) };
    let mut count = *nflush;

    while count > 0 {
        let line = remaining.take().unwrap();
        let next = line.l_next;
        if !line.l_line.is_empty() {
            flush_blanks();
            flush_line(&line);
        }
        if next.is_some() {
            unsafe { NBLANK_LINES += 1 };
        }
        remaining = next;
        count -= 1;
    }

    if let Some(ref mut l) = remaining {
        l.l_prev = None;
    }
    unsafe {
        LINES = remaining;
    }
    *nflush = count;
}
