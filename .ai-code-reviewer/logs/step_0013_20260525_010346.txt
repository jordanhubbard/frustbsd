const CPU_LEVEL_WHICH: c_uint = 0;

const SC_NPROCESSORS_CONF: c_int = 83;

#[repr(C)]
struct cpuset_t {
    bits: [u64; 16],
}

extern "C" {
    fn sysconf(name: c_int) -> c_long;
    fn cpuset_getaffinity(
        level: c_uint,

        id: c_int,
        setsize: usize,

    ) -> c_int;
}

fn cpu_zero(set: &mut cpuset_t) {
    for bit in set.bits.iter_mut() {
        *bit = 0;
    }
}

fn cpu_count(set: &cpuset_t) -> c_int {
    let mut count: c_int = 0;
    for bit in &set.bits {
        count += bit.count_ones() as c_int;
    }
    count
}
