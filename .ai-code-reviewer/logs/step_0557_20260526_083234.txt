use std::env;
use std::fs::File;
use std::io::{self, Read};
use std::process;

const NOTE_NAMES: &[(Option<&str>, &str)] = &[
    (None, "C"),
    (Some("Db"), "C#"),
    (None, "D"),
    (Some("Eb"), "D#"),
    (None, "E"),
    (None, "F"),
    (Some("Gb"), "F#"),
    (None, "G"),
    (Some("Ab"), "G#"),
    (None, "A"),
    (Some("Bb"), "A#"),
    (None, "B"),
];

const CHAN_MASK: u8 = 0x0f;

fn note2octave(n: u8) -> i32 {
    (n as i32) / 12 - 1
}

fn note2freq(n: u8) -> f64 {
    440.0 * 2.0f64.powf(((n as f64) - 69.0) / 12.0)
}

fn usage() -> ! {
    eprintln!("usage: {} [-t] device", env::args().next().unwrap_or_else(|| "mididump".to_string()));
    process::exit(1);
}

fn read_byte<R: Read>(reader: &mut R) -> u8 {
    let mut buf = [0u8; 1];
    match reader.read_exact(&mut buf) {
        Ok(()) => buf[0],
        Err(e) => {
            eprintln!("read: {}", e);
            process::exit(1);
        }
    }
}

fn main() {
    let args: Vec<String> = env::args().collect();
    let mut tflag = false;
    let mut device = None;

    let mut i = 1;
    while i < args.len() {
        if args[i] == "-t" {
            tflag = true;
            i += 1;
        } else if args[i].starts_with('-') {
            usage();
        } else {
            device = Some(&args[i]);
            i += 1;
        }
    }

    let device = match device {
        Some(d) => d,
        None => usage(),
    };

    let file = match File::open(device) {
        Ok(f) => f,
        Err(e) => {
            eprintln!("open({}): {}", device, e);
            process::exit(1);
        }
    };

    let mut reader = io::BufReader::new(file);

    loop {
        let event = read_byte(&mut reader);
        if event & 0x80 == 0 {
            continue;
        }
        let chan = ((event & CHAN_MASK) + 1) as u32;

        match event {
            0x80..=0x8f | 0x90..=0x9f => {
                let b1 = read_byte(&mut reader);
                let b2 = read_byte(&mut reader);
                let idx = (b1 % NOTE_NAMES.len() as u8) as usize;
                let (alt, name) = NOTE_NAMES[idx];
                let octave = note2octave(b1);
                let mut buf = format!("{}{}", name, octave);
                if let Some(alt_name) = alt {
                    buf.push_str(&format!("/{}{}", alt_name, octave));
                }
                let on_off = if event >= 0x80 && event <= 0x8f { "off" } else { "on" };
                println!(
                    "Note {:<3}\t\tchannel={}, note={} ({}, {:.2}Hz), velocity={}",
                    on_off, chan, b1, buf, note2freq(b1), b2
                );
            }
            0xa0..=0xaf => {
                let b1 = read_byte(&mut reader);
                let b2 = read_byte(&mut reader);
                println!(
                    "Polyphonic aftertouch\tchannel={}, note={}, pressure={}",
                    chan, b1, b2
                );
            }
            0xb0..=0xbf => {
                let b1 = read_byte(&mut reader);
                let b2 = read_byte(&mut reader);
                let ctl_name = match ctl_name(b1) {
                    Some(n) => n,
                    None => {
                        println!(
                            "Control/Mode change\tchannel={}, control={}, value={}",
                            chan, b1, b2
                        );
                        continue;
                    }
                };
                print!(
                    "Control/Mode change\tchannel={}, control={} ({}), value={}",
                    chan, b1, ctl_name, b2
                );

                        print!(" (off)");
                    } else {
                        print!(" (on)");
                    }

                    } else if b2 == 127 {
                        print!(" (on");
                    }
                }

            }
            0xc0..=0xcf => {
                let b1 = read_byte(&mut reader);
                println!("Program change\t\tchannel={}, program={}", chan, b1);
            }

                let b1 = read_byte(&mut reader);
                println!(
                    "Channel aftertouch\tchannel={}, pressure={}",
                    chan, b1
                );
            }
            0xe0..=0xef => {
                let b1 = read_byte(&mut reader);
                let b2 = read_byte(&mut reader);
                println!(
                    "Pitch bend\t\tchannel={}, change={}",
                    chan,
                    (b1 as i32) | ((b2 as i32) << 7)
                );
            }
            0xf0 => {

                print!("0x{:02x}", b1);

            0xf2 => {
                let b1 = read_byte(&mut reader);
                let b2 = read_byte(&mut reader);
                println!(
                    "Song position pointer\tptr={}",
                    (b1 as u32) | ((b2 as u32) << 7)
                );
            }

                let b1 = read_byte(&mut reader);
                println!("Song select\t\tsong={}", b1);
            }

            }
            0xf7 => {
