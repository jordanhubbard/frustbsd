use std::io::{self, Write};
use std::process;
use std::thread;
use std::time::Duration;

fn usage() -> ! {
    let msg = "usage: sleep number[unit] [...]\n\
        Unit can be 's' (seconds, the default), \
        m (minutes), h (hours), or d (days).\n";
    let _ = io::stderr().write_all(msg.as_bytes());
    process::exit(1);
}

fn parse_interval(arg: &str) -> Option<f64> {
    let mut chars = arg.chars();
    let unit_char = chars.next_back();
    let num_str: String = chars.collect();

    let num: f64 = match num_str.parse() {
        Ok(n) => n,
        Err(_) => {
            eprintln!("sleep: invalid time interval: {}", arg);
            return None;
        }
    };

    if num.is_nan() {
        eprintln!("sleep: invalid time interval: {}", arg);
        return None;
    }

    match unit_char {
        Some('d') => Some(num * 24.0 * 60.0 * 60.0),
        Some('h') => Some(num * 60.0 * 60.0),
        Some('m') => Some(num * 60.0),
        Some('s') | None => Some(num),
        Some(_) => {
            eprintln!("sleep: invalid time interval: {}", arg);
            None
        }
    }
}

fn main() {
    let args: Vec<String> = std::env::args().skip(1).collect();

    if args.is_empty() {
        usage();
    }

    let mut total_seconds: f64 = 0.0;
    for arg in &args {
        match parse_interval(arg) {
            Some(seconds) => total_seconds += seconds,
            None => usage(),
        }
    }

    if total_seconds > i64::MAX as f64 {
        usage();
    }

    if total_seconds < 1e-9 {
        process::exit(0);
    }

    let duration = Duration::from_secs_f64(total_seconds);
    thread::sleep(duration);
    process::exit(0);
}
