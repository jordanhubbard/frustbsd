// FreeBSD-compatible replacement for Windows OSUtil functions.
// Provides equivalent functionality using POSIX/FreeBSD syscalls.

use std::os::raw::c_char;

/// Exit the process with the given status code.
/// Corresponds to Windows ExitProcess(status).
#[no_mangle]
pub unsafe extern "C" fn __windows_osutil_exit(status: i32) {
    std::process::exit(status);
}

/// Write a message to stderr.
/// Corresponds to Windows WriteFile to STD_ERROR_HANDLE.
#[no_mangle]
pub unsafe extern "C" fn __windows_osutil_write_to_stderr(
    msg: *const c_char,
    len: usize,
) {
    if msg.is_null() || len == 0 {
        return;
    }
    let slice = std::slice::from_raw_parts(msg as *const u8, len);
    let _ = std::io::stderr().write_all(slice);
}
