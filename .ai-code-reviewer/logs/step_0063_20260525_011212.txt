const WHOIS_QUICK: u32 = 0x02;
const WHOIS_SPAM_ME: u32 = 0x04;

struct WhoisWhere {

}

fn usage() -> ! {
    eprintln!("usage: whois [-aAbfghIklmPrRS] [-c country] [-h host] [-p port] name ...");
    process::exit(1);
}

fn choose_server(domain: &str) -> &'static str {
    let len = domain.len();
    for entry in &WHOISWHERE {
        let suffix_len = entry.suffix.len();
        if len > suffix_len {
            if let Some(suffix) = domain.get(len - suffix_len..) {
                if suffix.eq_ignore_ascii_case(entry.suffix) {
                    return entry.server;
                }
            }
        }
    }
    eprintln!("whois: no default whois server");
    process::exit(70);
}

fn is_host_char(c: char) -> bool {
    c.is_ascii_alphanumeric() || c == '.' || c == '-'
}

fn whois(query: &str, hostname: &str, hostport: &str, flags: u32) {
    let addr_str = format!("{}:{}", hostname, hostport);
    let addrs = match addr_str.to_socket_addrs() {
        Ok(addrs) => addrs,
        Err(e) => {
            eprintln!("whois: {}: {}", hostname, e);
            process::exit(1);
        }
    };

    let mut stream = None;
    for addr in addrs {
        match TcpStream::connect_timeout(&addr, std::time::Duration::from_secs(180)) {
            Ok(s) => {
                stream = Some(s);
                break;
            }
            Err(_) => continue,
        }
    }

    let mut stream = match stream {
        Some(s) => s,
        None => {
            eprintln!("whois: connect(): Connection timed out");
            process::exit(1);
        }
    };

    let mut out = &mut stream as &mut dyn Write;
    let mut reader = BufReader::new(&stream);

    // Send query
    if (flags & WHOIS_SPAM_ME) == 0 &&
       (hostname.eq_ignore_ascii_case(DENICHOST) ||
        hostname.eq_ignore_ascii_case(&format!("de{}", QNICHOST_TAIL))) {
        let idn = query.chars().any(|c| !c.is_ascii());
        if idn {
            let _ = writeln!(out, "-T dn{} {}", query);
        } else {
            let _ = writeln!(out, "-T dn,ace {}", query);
        }
    } else if (flags & WHOIS_SPAM_ME) == 0 &&
              (hostname.eq_ignore_ascii_case(DKNICHOST) ||
               hostname.eq_ignore_ascii_case(&format!("dk{}", QNICHOST_TAIL))) {
        let _ = writeln!(out, "--show-handles {}", query);
    } else if (flags & WHOIS_SPAM_ME) != 0 || query.contains(' ') {
        let _ = writeln!(out, "{}", query);
    } else if hostname.eq_ignore_ascii_case(ANICHOST) {
        if query.starts_with("AS") && query[2..].chars().all(|c| c.is_ascii_digit()) {
            let _ = writeln!(out, "+ a {}", &query[2..]);
        } else {
            let _ = writeln!(out, "+ {}", query);
        }
    } else {
        let _ = writeln!(out, "{}", query);
    }
    let _ = out.flush();

    let mut comment: i32 = 0;
    if (flags & WHOIS_SPAM_ME) == 0 &&
       (hostname.eq_ignore_ascii_case(ANICHOST) ||
        hostname.eq_ignore_ascii_case(RNICHOST)) {
        comment = 2;
    }

    let mut nhost: Option<String> = None;
    let mut nport: Option<String> = None;

    loop {
        let mut line = String::new();
        match reader.read_line(&mut line) {
            Ok(0) => break,
            Ok(_) => {}
            Err(_) => break,
        }

        let len = line.len();

        // Nominet
        if (flags & WHOIS_SPAM_ME) == 0 && len == 5 && line == "-- \r\n" {
            break;
        }

        // RIRs
        if comment == 1 && !line.is_empty() && line.as_bytes()[0] == b'#' {
            break;
        } else if comment == 2 {
            if !line.is_empty() && "#%\r\n".contains(line.chars().next().unwrap()) {
                continue;
            } else {
                comment = 1;
            }
        }

        print!("{}", line);

        if (flags & WHOIS_RECURSE) != 0 && nhost.is_none() {
            for referral in &WHOIS_REFERRAL {
                if let Some(pos) = line.find(referral.prefix) {
                    let after_prefix = &line[pos + referral.prefix.len()..];
                    let trimmed = after_prefix.trim_start();
                    let host_end = trimmed.find(|c: char| !is_host_char(c)).unwrap_or(trimmed.len());
                    if host_end > 0 {
                        let host = &trimmed[..host_end];
                        if !host.is_empty() {
                            nhost = Some(host.to_string());
                            if trimmed.as_bytes().get(host_end) == Some(&b':') {
                                let port_str = &trimmed[host_end + 1..];
                                let port_end = port_str.find(|c: char| !c.is_ascii_digit()).unwrap_or(port_str.len());
                                if port_end > 0 && port_end < 6 {
                                    nport = Some(port_str[..port_end].to_string());
                                } else {
                                    nport = Some(hostport.to_string());
                                }
                            } else {
                                nport = Some(hostport.to_string());
                            }
                            break;
                        }
                    }
                }
            }

            if nhost.is_none() {
                for pattern in &ACTUALLY_ARIN {
                    if line.starts_with(*pattern) {
                        nhost = Some(ANICHOST.to_string());
                        nport = Some(hostport.to_string());
                        break;
                    }
                }
            }
        }

        // Verisign etc.
        if (flags & WHOIS_SPAM_ME) == 0 &&
           len >= CHOPSPAM.len() &&
           (line[..CHOPSPAM.len()].eq_ignore_ascii_case(CHOPSPAM) ||
            (len >= CHOPSPAM.len() - 4 &&
             line[4..CHOPSPAM.len()].eq_ignore_ascii_case(&CHOPSPAM[4..]))) {
            println!();
            break;
        }
    }

    // Track RIR visits
    let mut f = false;
    unsafe {
        for entry in TRY_RIR.iter_mut() {
            if !entry.loop && entry.host.eq_ignore_ascii_case(hostname) {
                entry.loop = true;
            }
        }

        for entry in TRY_RIR.iter() {
            if entry.loop && nhost.as_ref().map_or(false, |h| h.eq_ignore_ascii_case(entry.host)) {
                nhost = None;
                nport = None;
                f = true;
            }
        }

        if f {
            for entry in TRY_RIR.iter() {
                if !entry.loop {
                    nhost = Some(entry.host.to_string());
                    nport = Some(hostport.to_string());
                    break;
                }
            }
        }
    }

    if let Some(ref nh) = nhost {
        if !nh.eq_ignore_ascii_case(hostname) {
            println!("# {}\n", nh);
            let np = nport.as_ref().map(|s| s.as_str()).unwrap_or(hostport);
            whois(query, nh, np, flags);
        }
    }
}
