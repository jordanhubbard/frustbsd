struct Alloc {
    start: *mut std::ffi::c_void,
    size: usize,
}

impl Alloc {
    fn new() -> Self {
        Alloc { start: ptr::null_mut(), size: 0 }
    }

    fn is_empty(&self) -> bool {
        self.size == 0
    }

    fn realloc_buf(&mut self) {
        let old_size = self.size;
        let new_size = if old_size == 0 {
            ALLOC_FIRST_SIZE
        } else if old_size < PAGE_SIZE {
            2 * old_size
        } else {
            let aligned = (old_size + PAGE_SIZE - 1) & !(PAGE_SIZE - 1);
            aligned + PAGE_SIZE
        };

        let p = unsafe { realloc(self.start, new_size) };
        if p.is_null() {
            eprintln!("cannot realloc allocation (old size: {}, new: {})", old_size, new_size);
            process::exit(1);
        }
        self.start = p;
        self.size = new_size;
    }

    fn free_buf(&mut self) {
        if !self.is_empty() {
            unsafe { free(self.start) };
            self.start = ptr::null_mut();
            self.size = 0;
        }
    }
}

fn alloc_getpwnam(name: &str, alloc: &mut Alloc) -> Result<libc::passwd, i32> {
    let cname = CString::new(name).unwrap();
    let mut result: *mut libc::passwd = ptr::null_mut();

    if alloc.is_empty() {
        alloc.realloc_buf();
    }

    loop {
        let pwd_ptr = alloc.start as *mut libc::passwd;
        let buf_ptr = unsafe { (pwd_ptr as *mut u8).add(std::mem::size_of::<libc::passwd>()) };
        let buflen = alloc.size - std::mem::size_of::<libc::passwd>();

        let error = unsafe {
            getpwnam_r(
                cname.as_ptr(),
                pwd_ptr,
                buf_ptr as *mut c_char,
                buflen,
                &mut result,
            )
        };

        if error != 0 {
            if error == 34 { /* ERANGE */
                alloc.realloc_buf();
                continue;
            }
            return Err(error);
        }

        if result.is_null() {
            return Err(2); /* ENOENT */
        }

        return unsafe { Ok((*result).clone()) };
    }
}

fn parse_user_pwd(name: &str) -> Result<uid_t, String> {
    // Try as username first
    let mut alloc = Alloc::new();
    match alloc_getpwnam(name, &mut alloc) {
        Ok(pwd) => {
            alloc.free_buf();
            return Ok(pwd.pw_uid);
        }
        Err(2) => {} /* ENOENT - not found, try numeric */
        Err(e) => {
            alloc.free_buf();
            return Err(format!("cannot access the password database: error {}", e));
        }
    }

    // Try as numeric UID
    match name.parse::<u32>() {
        Ok(uid) => Ok(uid),
        Err(_) => Err(format!("invalid UID '{}'", name)),
    }
}

fn parse_group(name: &str) -> Result<gid_t, String> {
    let mut alloc = Alloc::new();
    match alloc_getgrnam(name, &mut alloc) {
        Ok(grp) => {
            alloc.free_buf();
            return Ok(grp.gr_gid);
        }
        Err(2) => {} /* ENOENT - not found, try numeric */
        Err(e) => {
            alloc.free_buf();
            return Err(format!("cannot access the group database: error {}", e));
        }
    }

    // Try as numeric GID
    match name.parse::<u32>() {
        Ok(gid) => Ok(gid),
        Err(_) => Err(format!("invalid GID '{}'", name)),
    }
}
