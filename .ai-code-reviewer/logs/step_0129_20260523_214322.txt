//! Rust replacement for the FreeBSD_version example program.
//!
//! Demonstrates how to determine the FreeBSD release version at both
//! compile time and runtime.

extern "C" {
    fn getosreldate() -> u32;
}

fn main() {
    // Compile-time FreeBSD version
    let compile_version = option_env!("FREEBSD_VERSION")
        .and_then(|v| v.parse::<u32>().ok())
        .unwrap_or(0);

    println!("Compilation release date: {}", compile_version);

    // Runtime FreeBSD version
    if compile_version >= 199408 {
        let runtime_version = unsafe { getosreldate() };
        println!("Execution environment release date: {}", runtime_version);
    } else {
        println!("Execution environment release date: can't tell");
    }
}
