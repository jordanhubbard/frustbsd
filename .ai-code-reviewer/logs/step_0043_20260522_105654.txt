fn main() {
    let args: Vec<String> = std::env::args().collect();

    let input: String;
    if args.len() < 2 {
        let msg = "When do you have to leave? ";
        io::stdout().write_all(msg.as_bytes()).unwrap();
        io::stdout().flush().unwrap();
        let mut buf = String::new();
        io::stdin().read_line(&mut buf).unwrap_or_default();
        input = buf;
        if input.is_empty() || input.trim() == "" {
            process::exit(0);
        }
    } else if args.len() > 2 {
        usage();
    } else {
        input = args[1].clone();
    }

    let mut cp = input.trim_end_matches('\n');

    let plusnow = if cp.starts_with('+') {
        cp = &cp[1..];
        true
    } else {
        false
    };

    if cp.is_empty() {
        usage();
    }

    for c in cp.chars() {
        if !c.is_ascii_digit() {
            usage();
        }
    }

    let num: u32 = cp.parse().unwrap_or_else(|_| usage());
    let minutes: u32 = num % 100;
    let hours: u32 = num / 100;

    if minutes > 59 {
        usage();
    }

    let secs: u64 = if plusnow {
        (hours as u64) * 3600 + (minutes as u64) * 60
    } else {
        if hours > 23 {
            usage();
        }

        let mut now: c_long = 0;
        unsafe {
            now = time(&mut now);
        }
        let now_secs = now as u64;

        let now_minutes = (now_secs / 60) % 60;
        let now_hours = (now_secs / 3600) % 24;

        let hours_12 = if hours > 11 { hours - 12 } else { hours };
        let now_hours_12 = if now_hours > 11 { now_hours - 12 } else { now_hours };

        let mut target_hours = hours_12;
        if hours_12 < now_hours_12 || (hours_12 == now_hours_12 && minutes <= now_minutes) {
            target_hours += 12;
        }

        let mut secs = (target_hours as u64 - now_hours_12 as u64) * 3600;
        secs += (minutes as u64 - now_minutes) * 60;
        secs -= now_secs % 60;
        secs
    };

    doalarm(secs);
    process::exit(0);
}

fn doalarm(secs: u64) {
    let pid = unsafe { fork() };

    if pid > 0 {
        let mut daytime: c_long = 0;
        unsafe {
            daytime = time(&mut daytime);
        }
        daytime += secs as c_long;

        let mut tb = [0u8; 80];
        unsafe {
            let tm_ptr = localtime(&daytime);
            let fmt = CString::new("%+").unwrap();
            let ret = strftime(
                tb.as_mut_ptr() as *mut c_char,
                tb.len(),
                fmt.as_ptr(),
                tm_ptr,
            );
            if ret > 0 {
                let s = String::from_utf8_lossy(&tb[..ret]);
                println!("Alarm set for {}. (pid {})", s, pid);
            } else {
                println!("Alarm set. (pid {})", pid);
            }
        }
        process::exit(0);
    } else if pid == 0 {
        unsafe {
            sleep(2);
        }
        let mut secs = if secs >= 2 { secs - 2 } else { 0 };

        const FIVEMIN: u64 = 5 * 60;
        const ONEMIN: u64 = 60;

        if secs >= FIVEMIN {
            unsafe {
                sleep((secs - FIVEMIN) as c_uint);
            }
            let msg = "\x07\x07You have to leave in 5 minutes.\n";
            if write_stdout(msg) {
                process::exit(0);
            }
            secs = FIVEMIN;
        }

        if secs >= ONEMIN {
            unsafe {
                sleep((secs - ONEMIN) as c_uint);
            }
            let msg = "\x07\x07Just one more minute!\n";
            if write_stdout(msg) {
                process::exit(0);
            }
        }

        for _ in 0..10 {
            unsafe {
                sleep(ONEMIN as c_uint);
            }
            let msg = "\x07\x07Time to leave!\n";
            if write_stdout(msg) {
                process::exit(0);
            }
        }

        let msg = "\x07\x07That was the last time I'll tell you.  Bye.\n";
        let _ = write_stdout(msg);
        process::exit(0);
    } else {
        eprintln!("fork failed");
        process::exit(1);
    }
}

fn write_stdout(msg: &str) -> bool {
    let fd = io::stdout().as_raw_fd();
    unsafe {
        let ret = write(fd, msg.as_ptr() as *const c_void, msg.len());
        ret < 0
    }
}
