use std::env;
use std::fs::File;
use std::io::{self, BufReader, Read, Write};
use std::process;

static MAX_TAB_STOPS: usize = 100;

fn usage() -> ! {
    eprintln!("usage: unexpand [-a | -t tablist] [file ...]");
    process::exit(1);
}

fn getstops(spec: &str) -> Vec<usize> {
    let mut stops: Vec<usize> = Vec::new();
    let mut cp = spec.as_bytes();

    loop {
        let mut i: usize = 0;
        while let Some(&b @ b'0'..=b'9') = cp.first() {
            i = i * 10 + (b - b'0') as usize;
            cp = &cp[1..];
        }
        if i == 0 {
            eprintln!("unexpand: bad tab stop spec");
            process::exit(1);
        }
        if let Some(&last) = stops.last() {
            if i <= last {
                eprintln!("unexpand: bad tab stop spec");
                process::exit(1);
            }
        }
        if stops.len() >= MAX_TAB_STOPS {
            eprintln!("unexpand: too many tab stops");
            process::exit(1);
        }
        stops.push(i);

        if cp.is_empty() {
            break;
        }
        if cp[0] != b',' && !matches!(cp[0], b' ' | b'\t') {
            eprintln!("unexpand: bad tab stop spec");
            process::exit(1);
        }
        cp = &cp[1..];
    }

    stops
}

fn wcwidth(ch: char) -> usize {
    if ch == '\0' {
        return 0;
    }
    if ch.is_ascii_graphic() || ch == ' ' {
        return 1;
    }
    let cp = ch as u32;
    if (0x1100..=0x115f).contains(&cp)
        || (0x2e80..=0xa4cf).contains(&cp)
        || (0xac00..=0xd7a3).contains(&cp)
        || (0xf900..=0xfaff).contains(&cp)
        || (0xfe10..=0xfe19).contains(&cp)
        || (0xfe30..=0xfe6f).contains(&cp)
        || (0xff00..=0xff60).contains(&cp)
        || (0xffe0..=0xffe6).contains(&cp)
        || (0x20000..=0x2fffd).contains(&cp)
        || (0x30000..=0x3fffd).contains(&cp)
    {
        return 2;
    }
    if (0x0300..=0x036f).contains(&cp)
        || (0x1ab0..=0x1aff).contains(&cp)
        || (0x1dc0..=0x1dff).contains(&cp)
        || (0x20d0..=0x20ff).contains(&cp)
        || (0xfe20..=0xfe2f).contains(&cp)
    {
        return 0;
    }
    1
}

fn tabify<R: Read>(
    mut reader: BufReader<R>,
    curfile: &str,
    all: bool,
    tabstops: &[usize],
) -> bool {
    let mut buf = String::new();
    if reader.read_to_string(&mut buf).is_err() {
        eprintln!("unexpand: {}: I/O error", curfile);
        return false;
    }

    let stdout = io::stdout();
    let mut out = stdout.lock();
    let nstops = tabstops.len();
    let limit = if nstops == 1 {
        usize::MAX
    } else {
        *tabstops.last().unwrap() - 1
    };

    let chars: Vec<char> = buf.chars().collect();
    let mut i = 0;
    let mut failed = false;

    while i < chars.len() {
        let mut doneline = false;
        let mut ocol: usize = 0;
        let mut dcol: usize = 0;

        while i < chars.len() {
            let ch = chars[i];

            if ch == ' ' && !doneline {
                dcol += 1;
                if dcol >= limit {
                    doneline = true;
                }
                i += 1;
                continue;
            } else if ch == '\t' {
                if nstops == 1 {
                    let ts = tabstops[0];
                    dcol = (1 + dcol / ts) * ts;
                    i += 1;
                    continue;
                } else {
                    let mut n = 0;
                    while n < nstops && tabstops[n] - 1 < dcol {
                        n += 1;
                    }
                    if n < nstops - 1 && tabstops[n] - 1 < limit {
                        dcol = tabstops[n];
                        i += 1;
                        continue;
                    }
                    doneline = true;
                }
            }

            // Output maximal number of tabs.
            if nstops == 1 {
                let ts = tabstops[0];
                while (ocol + ts) / ts <= dcol / ts {
                    if dcol - ocol < 2 {
                        break;
                    }
                    if write!(out, "\t").is_err() {
                        failed = true;
                        break;
                    }
                    ocol = (1 + ocol / ts) * ts;
                }
            } else {
                let mut n = 0;
                while n < nstops && tabstops[n] - 1 < ocol {
                    n += 1;
                }
                while ocol < dcol && n < nstops && ocol < limit {
                    if write!(out, "\t").is_err() {
                        failed = true;
                        break;
                    }
                    ocol = tabstops[n];
                    n += 1;
                }
            }
            if failed {
                break;
            }

            // Then spaces.
            while ocol < dcol && ocol < limit {
                if write!(out, " ").is_err() {
                    failed = true;
                    break;
                }
                ocol += 1;
            }
            if failed {
                break;
            }

            if ch == '\b' {
                if write!(out, "\b").is_err() {
                    failed = true;
                } else if ocol > 0 {
                    ocol -= 1;
                    dcol -= 1;
                }
            } else if ch == '\n' {
                if write!(out, "\n").is_err() {
                    failed = true;
                }
                doneline = false;
                ocol = 0;
                dcol = 0;
                i += 1;
                continue;
            } else if ch != ' ' || dcol > limit {
                if write!(out, "{}", ch).is_err() {
                    failed = true;
                } else {
                    let width = wcwidth(ch);
                    if width > 0 {
                        ocol += width;
                        dcol += width;
                    }
                }
            }
            i += 1;

            if failed {
                break;
            }

            // Only processing leading blanks or we've gone past the
            // last tab stop. Emit remainder of this line unchanged.
            if !all || dcol >= limit {
                while i < chars.len() && chars[i] != '\n' {
                    if write!(out, "{}", chars[i]).is_err() {
                        failed = true;
                        break;
                    }
                    i += 1;
                }
                if i < chars.len() && chars[i] == '\n' {
                    if write!(out, "\n").is_err() {
                        failed = true;
                    }
                    i += 1;
                }
                doneline = false;
                ocol = 0;
                dcol = 0;
                break;
            }
        }
        if failed {
            break;
        }
    }

    if failed {
        eprintln!("unexpand: {}: I/O error", curfile);
        return false;
    }
    true
}

fn main() {
    let args: Vec<String> = env::args().collect();
    let mut all = false;
    let mut tabstops: Vec<usize> = vec![8];

    let mut i = 1;
    let mut filenames: Vec<String> = Vec::new();

    while i < args.len() {
        if args[i] == "-a" {
            all = true;
            i += 1;
        } else if args[i].starts_with("-t") {
            let spec = if args[i].len() > 2 {
                &args[i][2..]
            } else if i + 1 < args.len() {
                i += 1;
                &args[i]
            } else {
                usage();
            };
            tabstops = getstops(spec);
            all = true;
            i += 1;
        } else if args[i].starts_with('-') {
            usage();
        } else {
            filenames.push(args[i].clone());
            i += 1;
        }
    }

    let mut failed = false;

    if filenames.is_empty() {
        let stdin = io::stdin();
        let reader = BufReader::new(stdin.lock());
        if !tabify(reader, "stdin", all, &tabstops) {
            failed = true;
        }
    } else {
        for filename in &filenames {
            match File::open(filename) {
                Ok(f) => {
                    let reader = BufReader::new(f);
                    if !tabify(reader, filename, all, &tabstops) {
                        failed = true;
                    }
                }
                Err(e) => {
                    eprintln!("unexpand: {}: {}", filename, e);
                    failed = true;
                }
            }
        }
    }

    if failed {
        process::exit(1);
    }
}
