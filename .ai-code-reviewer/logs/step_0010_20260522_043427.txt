extern "C" {
    fn yp_get_default_domain(domain: *mut *mut u8) -> i32;
    fn yp_match(
        domain: *const u8,
        map: *const u8,
        key: *const u8,
        keylen: i32,
        value: *mut *mut u8,
        valuelen: *mut i32,
    ) -> i32;
    fn yperr_string(err: i32) -> *const u8;
    fn xdr_free(xdr_op: u64, obj: *mut u8);
}

const YPERR_YPBIND: i32 = 14;

struct YpAlias {
    alias: &'static str,
    name: &'static str,
}

const YPALIASES: &[YpAlias] = &[
    YpAlias { alias: "passwd", name: "passwd.byname" },
    YpAlias { alias: "master.passwd", name: "master.passwd.byname" },
    YpAlias { alias: "shadow", name: "shadow.byname" },
    YpAlias { alias: "group", name: "group.byname" },
    YpAlias { alias: "networks", name: "networks.byaddr" },
    YpAlias { alias: "hosts", name: "hosts.byname" },
    YpAlias { alias: "protocols", name: "protocols.bynumber" },
    YpAlias { alias: "services", name: "services.byname" },
    YpAlias { alias: "aliases", name: "mail.aliases" },
    YpAlias { alias: "ethers", name: "ethers.byname" },
];

fn usage() -> ! {
    eprintln!(
        "{}\n{}",
        "usage: ypmatch [-kt] [-d domainname] key ... mapname",
        "       ypmatch -x"
    );
    std::process::exit(1);
}

fn main() {
    let args: Vec<String> = std::env::args().collect();
    let mut domainname: Option<Vec<u8>> = None;
    let mut notrans: bool = false;
    let mut key: bool = false;
    let mut show_aliases: bool = false;

    let mut i = 1;
    while i < args.len() {
        let arg = &args[i];
        if arg == "-x" {
            show_aliases = true;
            i += 1;
        } else if arg.starts_with("-d") {
            if arg.len() > 2 {
                domainname = Some(arg[2..].as_bytes().to_vec());
            } else if i + 1 < args.len() {
                i += 1;
                domainname = Some(args[i].as_bytes().to_vec());
            } else {
                usage();
            }
            i += 1;
        } else if arg == "-t" {
            notrans = true;
            i += 1;
        } else if arg == "-k" {
            key = true;
            i += 1;
        } else if arg.starts_with('-') {
            usage();
        } else {
            break;
        }
    }

    if show_aliases {
        for a in YPALIASES {
            println!("Use \"{}\" for \"{}\"", a.alias, a.name);
        }
        std::process::exit(0);
    }

    if args.len() - i < 2 {
        usage();
    }

    let mut domain_ptr: *mut u8 = std::ptr::null_mut();
    if domainname.is_none() {
        let rc = unsafe { yp_get_default_domain(&mut domain_ptr) };
        if rc != 0 || domain_ptr.is_null() {
            eprintln!("can't get default domain");
            std::process::exit(1);
        }
        domainname = Some(unsafe {
            let mut v = Vec::new();
            let mut p = domain_ptr;
            while *p != 0 {
                v.push(*p);
                p = p.add(1);
            }
            v
        });
        unsafe {
            xdr_free(0, domain_ptr);
        }
    }

    let domain_bytes = domainname.unwrap();
    let inmap = &args[args.len() - 1];
    let mut map_name = inmap.as_str();
    if !notrans {
        for a in YPALIASES {
            if inmap == a.alias {
                map_name = a.name;
                break;
            }
        }
    }

    let mut rval: i32 = 0;
    for key_arg in &args[i..args.len() - 1] {
        let inkey = key_arg.as_bytes();
        let mut outbuf: *mut u8 = std::ptr::null_mut();
        let mut outbuflen: i32 = 0;

        let r = unsafe {
            yp_match(
                domain_bytes.as_ptr(),
                map_name.as_ptr(),
                inkey.as_ptr(),
                inkey.len() as i32,
                &mut outbuf,
                &mut outbuflen,
            )
        };

        match r {
            0 => {
                if key {
                    print!("{}: ", key_arg);
                }
                let slice = unsafe { std::slice::from_raw_parts(outbuf, outbuflen as usize) };
                let s = std::str::from_utf8(slice).unwrap_or("<invalid utf8>");
                println!("{}", s);
                unsafe {
                    xdr_free(0, outbuf);
                }
            }
            YPERR_YPBIND => {
                eprintln!("not running ypbind");
                std::process::exit(1);
            }
            _ => {
                let reason = unsafe {
                    let ptr = yperr_string(r);
                    if ptr.is_null() {
                        "unknown error"
                    } else {
                        std::ffi::CStr::from_ptr(ptr).to_string_lossy()
                    }
                };
                eprintln!(
                    "can't match key {} in map {}. reason: {}",
                    key_arg, map_name, reason
                );
                rval = 1;
            }
        }
    }
    std::process::exit(rval);
}
