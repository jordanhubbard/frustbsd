fn xnum(s: &str) -> Option<i64> {
    let s = s.trim();
    if s.is_empty() { return None; }
    let (d, r) = if s.starts_with("0x") || s.starts_with("0X") { (&s[2..], 16) }
    else if s.starts_with('0') && s.len() > 1 { (s, 8) }
    else { (s, 10) };
    if d.is_empty() { return None; }
    i64::from_str_radix(d, r).ok()
}

fn parse_skip(s: &str) -> Option<(i64, i64)> {

}

fn skip_r<R: Read>(mut r: R, skip: i64, file: &str, f: &F) -> i32 {
