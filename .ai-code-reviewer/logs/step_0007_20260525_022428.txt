use std::env;
use std::io::{self, Write};
use std::process;

fn main() {
    let args: Vec<String> = env::args().skip(1).collect();

    let (mut nflag, start) = if !args.is_empty() && args[0] == "-n" {
        (true, 1)
    } else {
        (false, 0)
    };

    let stdout = io::stdout();
    let mut handle = stdout.lock();

    for (i, arg) in args.iter().enumerate().skip(start) {
        let arg = &arg;
        let bytes = arg.as_bytes();

        // Check for trailing \c on the last argument
        let (out_bytes, suppress_newline) = if i == args.len() - 1 - start
            && bytes.len() >= 2
            && bytes[bytes.len() - 2] == b'\\'
            && bytes[bytes.len() - 1] == b'c'
        {
            (&bytes[..bytes.len() - 2], true)
        } else {
            (bytes, false)
        };

        if suppress_newline {
            nflag = true;
        }

        if i > start {
            let _ = handle.write_all(b" ");
        }
        let _ = handle.write_all(out_bytes);
    }

    if !nflag {
        let _ = handle.write_all(b"\n");
    }

    if let Err(e) = handle.flush() {
        eprintln!("echo: flush: {}", e);
        process::exit(1);
    }
}
