fn ttyconv(arg: &str) -> String {
    if arg.len() == 2 {
        if arg == "co" {
            return "console".to_string();
        }
        return format!("tty{}", arg);
    }
    if let Some(stripped) = arg.strip_prefix("/dev/") {
        return stripped.to_string();
    }
    arg.to_string()
}

fn dateconv(arg: &str) -> i64 {
    let mut timet: i64 = 0;
    unsafe { time(&mut timet) };

    let mut t: tm = unsafe { std::mem::zeroed() };
    unsafe { localtime_r(&timet, &mut t) };

    let (main_part, secs) = if let Some(dot_pos) = arg.find('.') {
        let main = &arg[..dot_pos];
        let sec_str = &arg[dot_pos + 1..];
        if sec_str.len() != 2 {
            eprintln!("out of range or illegal time specification: [[CC]YY]MMDDhhmm[.SS]");
            process::exit(1);
        }
        let sec: i32 = sec_str.parse().unwrap_or(0);
        (main, sec)
    } else {
        (arg, 0)
    };

    t.tm_sec = secs;

    let len = main_part.len();
    let mut yearset = false;
    let mut p = 0;

    match len {
        12 => {
            // CCYYMMDDhhmm
            let cc: i32 = main_part[p..p + 2].parse().unwrap_or(0);
            p += 2;
            t.tm_year = cc * 100;
            yearset = true;
            // fall through to 10
        }
        _ => {}
    }

    if len >= 10 {
        let yy: i32 = main_part[p..p + 2].parse().unwrap_or(0);
        p += 2;
        if yearset {
            t.tm_year += yy;
        } else {
            if yy < 69 {
                t.tm_year = yy + 2000;
            } else {
                t.tm_year = yy + 1900;
            }
        }
        t.tm_year -= 1900;
    }

    if len >= 8 {
        t.tm_mon = main_part[p..p + 2].parse().unwrap_or(1) - 1;
        p += 2;
        t.tm_mday = main_part[p..p + 2].parse().unwrap_or(1);
        p += 2;
        t.tm_hour = main_part[p..p + 2].parse().unwrap_or(0);
        p += 2;
        t.tm_min = main_part[p..p + 2].parse().unwrap_or(0);
    } else if len == 4 {
        t.tm_hour = main_part[p..p + 2].parse().unwrap_or(0);
        p += 2;
        t.tm_min = main_part[p..p + 2].parse().unwrap_or(0);
    } else {
        eprintln!("out of range or illegal time specification: [[CC]YY]MMDDhhmm[.SS]");
        process::exit(1);
    }

    t.tm_isdst = -1;
    let result = unsafe { mktime(&mut t) };
    if result == -1 {
        eprintln!("out of range or illegal time specification: [[CC]YY]MMDDhhmm[.SS]");
        process::exit(1);
    }
    result
}

fn format_time(t: i64, d_first: bool, yflag: bool) -> String {
    let mut tm: tm = unsafe { std::mem::zeroed() };
    unsafe { localtime_r(&t, &mut tm) };

    let fmt = if d_first {
        if yflag {
            b"%a %e %b %Y %R"
        } else {
            b"%a %e %b %R"
        }
    } else {
        if yflag {
            b"%a %b %e %Y %R"
        } else {
            b"%a %b %e %R"
        }
    };

    let mut buf = [0u8; 80];
    let n = unsafe { strftime(buf.as_mut_ptr(), buf.len(), fmt.as_ptr(), &tm) };
    String::from_utf8_lossy(&buf[..n]).to_string()
}

fn format_time_short(t: i64) -> String {
    let mut tm: tm = unsafe { std::mem::zeroed() };
    unsafe { localtime_r(&t, &mut tm) };

    let fmt = b"%R";
    let mut buf = [0u8; 80];
    let n = unsafe { strftime(buf.as_mut_ptr(), buf.len(), fmt.as_ptr(), &tm) };
    String::from_utf8_lossy(&buf[..n]).to_string()
}

fn format_duration(delta: i64, width: usize) -> String {
    let mut tm: tm = unsafe { std::mem::zeroed() };
    unsafe { gmtime_r(&delta, &mut tm) };

    let fmt = if width >= 8 { b"%T" } else { b"%R" };
    let mut buf = [0u8; 80];
    let n = unsafe { strftime(buf.as_mut_ptr(), buf.len(), fmt.as_ptr(), &tm) };
    let time_str = String::from_utf8_lossy(&buf[..n]).to_string();

    if delta < 86400 {
        format!("({})", time_str)
    } else {
        format!(" ({}+{})", delta / 86400, time_str)
    }
}

                    .trim_end_matches('\0')
