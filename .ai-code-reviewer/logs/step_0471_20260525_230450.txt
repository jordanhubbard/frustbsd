// We need to read fs_magic from the superblock pointer libufs gives us.
// The superblock struct starts with fs_magic at offset 0.
unsafe fn get_fs_magic(fs_ptr: *const c_void) -> u32 {
    (fs_ptr as *const u32).read()

fn usage() -> ! {
    eprintln!("usage: clri special_device inode_number ...");
    process::exit(1);
}

fn main() {
    let args: Vec<String> = env::args().collect();
    if args.len() < 3 {
        usage();
    }

    let fsname = &args[1];
    let inodes = &args[2..];

    let c_fsname = match CString::new(fsname.as_bytes()) {
        Ok(s) => s,
        Err(_) => {
            eprintln!("Invalid filesystem path");
            process::exit(1);
        }
    };

    // libufs uses malloc internally for uufsd, so we must use a raw pointer
    let mut disk: *mut uufsd = std::ptr::null_mut();

    if unsafe { ufs_disk_fillout(&mut disk, c_fsname.as_ptr()) } == -1 {
        let err = unsafe {
            if !disk.is_null() {
                let eptr = ufs_disk_error(disk);
                if !eptr.is_null() {
                    CStr::from_ptr(eptr).to_string_lossy().into_owned()
                } else {
                    "unknown error".to_string()
                }
            } else {
                "unknown error".to_string()
            }
        };
        eprintln!("loading superblock: {}", err);
        process::exit(1);
    }

    let mut exitval: i32 = 0;

    for inode_str in inodes {
        let inonum: i32 = match inode_str.parse() {
            Ok(n) => n,
            Err(_) => {
                eprintln!("{} is not a valid inode number", inode_str);
                exitval = 1;
                continue;
            }
        };

        if inonum < UFS_ROOTINO {
            eprintln!("{} is not a valid inode number", inode_str);
            exitval = 1;
            continue;
        }

        println!("clearing {}", inonum);

        let mut dp: union_dinodep = unsafe { std::mem::zeroed() };

        if unsafe { getinode(disk, &mut dp, inonum) } == -1 {
            let err = unsafe {
                let eptr = ufs_disk_error(disk);
                if !eptr.is_null() {
                    CStr::from_ptr(eptr).to_string_lossy().into_owned()
                } else {
                    "unknown error".to_string()
                }
            };
            eprintln!("getinode: {}", err);
            exitval = 1;
            continue;
        }

        unsafe {
            let fs_ptr = ufs_disk_fs(disk);
            let fs_magic = get_fs_magic(fs_ptr);
            if fs_magic == FS_UFS1_MAGIC {

                let gen = (*dp.dp2).di_gen + 1;
                std::ptr::write_bytes(dp.dp2, 0, 1);
                (*dp.dp2).di_gen = gen;
            }
            putinode(disk);
            fsync(ufs_disk_fd(disk));
        }
    }

    unsafe { ufs_disk_close(disk); }
    process::exit(exitval);
}
