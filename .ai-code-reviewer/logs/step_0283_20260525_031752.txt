fn search(progname: &str) -> i32 {
    const KNOWN: &[(&str, i32)] = &[
        ("uuencode", 0),
        ("uudecode", 1),
        ("b64encode", 2),
        ("b64decode", 3),
        ("base64", 4),
        ("qp", 5),
    ];
    for &(name, idx) in KNOWN {
        if progname == name {
            return idx;
        }
    }
    -1
}

fn enc(c: u8) -> u8 {
    if c == 0 { '`' } else { (c & 0x7F) + b' ' }
}

fn uuencode(input: &mut dyn Read, output: &mut dyn Write, mode: u16, remotefile: String, raw: bool) {
    if !raw {
        writeln!(output, "begin {:o} {}", mode, remotefile).unwrap();
    }

    let mut buf = [0u8; 45];
    loop {
        let n = match input.read(&mut buf) {
            Ok(0) => break,
            Ok(n) => n,
            Err(_) => {
                eprintln!("read error");
                process::exit(1);
            }
        };

        write!(output, "{}", enc(n as u8 as u8)).unwrap();
        let mut p = 0;
        while p < n {
            let b0 = buf[p];
            let b1 = if p + 1 < n { buf[p + 1] } else { 0 };
            let b2 = if p + 2 < n { buf[p + 2] } else { 0 };

            write!(output, "{}", enc((b0 >> 2) as u8)).unwrap();
            write!(output, "{}", enc(((b0 << 4) | (b1 >> 4)) as u8)).unwrap();
            write!(output, "{}", enc(((b1 << 2) | (b2 >> 6)) as u8)).unwrap();
            write!(output, "{}", enc(b2 & 0x3F)).unwrap();
            p += 3;
        }
        writeln!(output).unwrap();
    }

    if !raw {
        writeln!(output, "{}end", enc(0)).unwrap();
    }
}

fn base64_encode(input: &mut dyn Read, output: &mut dyn Write, mode: u16, remotefile: String, raw: bool, columns: usize) {
    if !raw {
        writeln!(output, "begin-base64 {:o} {}", mode, remotefile).unwrap();
    }

    let b64chars = b"ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789+/";
    let mut buf = [0u8; 1023];
    let mut carry = 0usize;

    loop {
        let n = match input.read(&mut buf) {
            Ok(0) => break,
            Ok(n) => n,
            Err(_) => {
                eprintln!("read error");
                process::exit(1);
            }
        };

        let mut out = Vec::new();
        for chunk in buf[..n].chunks(3) {
            let b0 = chunk[0] as u32;
            let b1 = if chunk.len() > 1 { chunk[1] as u32 } else { 0 };
            let b2 = if chunk.len() > 2 { chunk[2] as u32 } else { 0 };

            out.push(b64chars[((b0 >> 2) & 0x3F) as usize]);
            out.push(b64chars[((b0 << 4) | (b1 >> 4)) as u32 & 0x3F] as u8);
            if chunk.len() > 1 {
                out.push(b64chars[((b1 << 2) | (b2 >> 6)) as u32 & 0x3F] as u8);
            }
            if chunk.len() > 2 {
                out.push(b64chars[(b2 & 0x3F) as usize]);
            }
        }

        if columns == 0 {
            output.write_all(&out).unwrap();
        } else {
            let mut i = 0;
            while i < out.len() {
                let remaining = out.len() - i;
                let to_write = (columns - carry).min(remaining);
                output.write_all(&out[i..i + to_write]).unwrap();
                carry = (carry + to_write) % columns;
                i += to_write;
                if carry == 0 {
                    writeln!(output).unwrap();
                }
            }
        }
    }

    if columns == 0 || carry != 0 {
        writeln!(output).unwrap();
    }
    if !raw {
        writeln!(output, "====").unwrap();
    }
}

fn dec(c: u8) -> u8 {
    if c == b'`' { 0 } else { c - b' ' }
}

fn base64_decode_char(c: u8) -> u8 {
    match c {
        b'A'..=b'Z' => c - b'A',
        b'a'..=b'z' => c - b'a' + 26,
        b'0'..=b'9' => c - b'0' + 52,
        b'+' => 62,
        b'/' => 63,
        _ => 0,
    }
}
