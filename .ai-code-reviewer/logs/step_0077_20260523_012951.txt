fn rotr(x: u64, n: u32) -> u64 {
    x.rotate_right(n)
}

fn ch(x: u64, y: u64, z: u64) -> u64 {
    (x & y) ^ (!x & z)
}

fn maj(x: u64, y: u64, z: u64) -> u64 {
    (x & y) ^ (x & z) ^ (y & z)
}

fn sigma0(x: u64) -> u64 {
    rotr(x, 28) ^ rotr(x, 34) ^ rotr(x, 39)
}

fn sigma1(x: u64) -> u64 {
    rotr(x, 14) ^ rotr(x, 18) ^ rotr(x, 41)
}

fn gamma0(x: u64) -> u64 {
    rotr(x, 1) ^ rotr(x, 8) ^ (x >> 7)

fn gamma1(x: u64) -> u64 {
    rotr(x, 19) ^ rotr(x, 61) ^ (x >> 6)

fn sha512(data: &[u8]) -> [u8; 64] {
    let mut h = INITIAL_HASH;
    let msg_len = data.len() as u64;
    let padded_len = ((data.len() + 1 + 127) / 128 + 1) * 128;
    let mut padded = vec![0u8; padded_len];
    padded[..data.len()].copy_from_slice(data);
    padded[data.len()] = 0x80;
    let bit_len = msg_len * 8;
    padded[padded.len() - 8..].copy_from_slice(&bit_len.to_be_bytes());

    for chunk in padded.chunks(128) {
        let block: [u8; 128] = chunk.try_into().unwrap();
        sha512_compress(&mut h, &block);
    }

    let mut out = [0u8; 64];
    for (i, &val) in h.iter().enumerate() {
        out[i * 8..(i + 1) * 8].copy_from_slice(&val.to_be_bytes());
    }
    out
}

pub unsafe extern "C" fn crypto_hash(
