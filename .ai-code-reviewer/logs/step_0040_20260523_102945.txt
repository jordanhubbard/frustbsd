static void
flush_lines(int nflush)
{
	LINE *l;

	while (--nflush >= 0) {
		l = lines;
		lines = l->l_next;
		if (l->l_line) {
			flush_blanks();
			flush_line(l);
			free(l->l_line);
		}
		if (l->l_next)
			nblank_lines++;
		free_line(l);
	}
	if (lines)
		lines->l_prev = NULL;
}
