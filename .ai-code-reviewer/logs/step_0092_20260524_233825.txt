fn clock_gettime(id: c_int) -> Result<libc::timespec, std::io::Error> {
    let mut ts: libc::timespec = unsafe { std::mem::zeroed() };
    let ret = unsafe { libc::clock_gettime(id, &mut ts) };
    if ret != 0 {
        Err(std::io::Error::last_os_error())
    } else {
        Ok(ts)
    }
}

fn getstathz() -> Result<i32, std::io::Error> {
    let mib = [ffi::CTL_KERN, ffi::KERN_CLOCKRATE];
    let mut clockrate: libc::clockinfo = unsafe { std::mem::zeroed() };
    let mut size: usize = std::mem::size_of::<libc::clockinfo>();
    let ret = unsafe {
        libc::sysctl(
            mib.as_ptr(),
            2,
            &mut clockrate as *mut _ as *mut _,
            &mut size,
            std::ptr::null(),
            0,
        )
    };
    if ret == -1 {
        Err(std::io::Error::last_os_error())
    } else {
        Ok(clockrate.stathz)
    }
}

fn showtime(out: &mut dyn Write, before: &libc::timespec, after: &libc::timespec,
    ru: &libc::rusage, hflag: bool, pflag: bool, lflag: bool) {

    elapsed.tv_nsec -= before.tv_nsec;

        elapsed.tv_nsec += 1_000_000_000;
    }

    let dp = unsafe { DECIMAL_POINT as char };

        writeln!(out, "user {}{}{:02}", ru.ru_utime.tv_sec, dp, ru.ru_utime.tv_usec / 10_000).ok();
        writeln!(out, "sys {}{}{:02}", ru.ru_stime.tv_sec, dp, ru.ru_stime.tv_usec / 10_000).ok();
    } else if hflag {
        humantime(out, elapsed.tv_sec, elapsed.tv_nsec / 10_000_000);
        write!(out, " real\t").ok();
        humantime(out, ru.ru_utime.tv_sec, ru.ru_utime.tv_usec / 10_000);
        write!(out, " user\t").ok();
        humantime(out, ru.ru_stime.tv_sec, ru.ru_stime.tv_usec / 10_000);
        writeln!(out, " sys").ok();
    } else {
        write!(out, "{:>9}{}{:02} real ", elapsed.tv_sec, dp, elapsed.tv_nsec / 10_000_000).ok();
        write!(out, "{:>9}{}{:02} user ", ru.ru_utime.tv_sec, dp, ru.ru_utime.tv_usec / 10_000).ok();
        writeln!(out, "{:>9}{}{:02} sys", ru.ru_stime.tv_sec, dp, ru.ru_stime.tv_usec / 10_000).ok();
    }
    let _ = lflag;
}

fn humantime(out: &mut dyn Write, sec: i64, centisec: i64) {
    let dp = unsafe { DECIMAL_POINT as char };
    let mut s = sec;
    let days = s / (60 * 60 * 24);
    s %= 60 * 60 * 24;
    let hrs = s / (60 * 60);
    s %= 60 * 60;
    let mins = s / 60;
    s %= 60;

    write!(out, "\t").ok();
    if days != 0 { write!(out, "{}d", days).ok(); }
    if hrs != 0 { write!(out, "{}h", hrs).ok(); }
    if mins != 0 { write!(out, "{}m", mins).ok(); }
    write!(out, "{}{}{:02}s", s, dp, centisec).ok();
}

unsafe fn fork() -> Result<i32, std::io::Error> {
    let pid = libc::fork();
    if pid == -1 {
        Err(std::io::Error::last_os_error())
    } else {
        Ok(pid)
    }
}
