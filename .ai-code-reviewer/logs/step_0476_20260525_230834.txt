/*
 * Allocate and zero a uufsd struct.
 * Returns pointer or NULL on failure.
 */
struct uufsd *
clri_alloc_disk(void)
{
	struct uufsd *disk;

	disk = calloc(1, sizeof(*disk));
	return (disk);
}

/*
 * Clear a dinode, preserving the generation number + 1.
 * Returns 0 on success, -1 on error.
 */
int
clri_clear_inode(struct uufsd *disk, union dinodep *dp)
{
	uint32_t magic;

	magic = disk->d_fs.fs_magic;
	if (magic == FS_UFS1_MAGIC) {
		uint32_t gen;

		gen = dp->dp1->di_gen + 1;
		memset(dp->dp1, 0, sizeof(*dp->dp1));
		dp->dp1->di_gen = gen;
	} else {
		uint64_t gen;

		gen = dp->dp2->di_gen + 1;
		memset(dp->dp2, 0, sizeof(*dp->dp2));
		dp->dp2->di_gen = gen;
	}
	if (putinode(disk) == -1)
		return (-1);
	if (fsync(disk->d_fd) == -1)

	return (0);
}
