use std::process;

fn main() {
    let compile_version: u32 = match option_env!("FREEBSD_VERSION") {
        Some(v) => match v.parse() {
            Ok(n) => n,
            Err(_) => 0,
        },
        None => 0,
    };

    println!("Compilation release date: {}", compile_version);

    let exec_version = getosreldate();
    if compile_version >= 199408 {
        println!("Execution environment release date: {}", exec_version);
    } else {
        println!("Execution environment release date: can't tell");
    }
}

fn getosreldate() -> u32 {
    // FreeBSD sysctl kern.osreldate
    let name = b"kern.osreldate\0";
    let mut value: u32 = 0;
    let mut len = std::mem::size_of::<u32>() as libc::size_t;

    let ret = unsafe {
        libc::sysctl(
            name.as_ptr() as *const _,
            1,
            &mut value as *mut _ as *mut _,
            &mut len,
            std::ptr::null(),
            0,
        )
    };

    if ret == -1 {
        0
    } else {
        value
    }
}
