fn doalarm(secs: u64) {
    let pid = unsafe { libc::fork() };

    if pid > 0 {
        let now = SystemTime::now()
            .duration_since(SystemTime::UNIX_EPOCH)
            .unwrap();
        let alarm_time = now + Duration::from_secs(secs);

        // Format time using strftime-like format
        let ts = chrono_like_format(alarm_time);
        println!("Alarm set for {}. (pid {})", ts, pid);
        process::exit(0);
    } else if pid == 0 {
        // Child process
        std::thread::sleep(Duration::from_secs(2));
        let mut secs = if secs >= 2 { secs - 2 } else { 0 };

        const FIVEMIN: u64 = 5 * 60;
        const ONEMIN: u64 = 60;

        if secs >= FIVEMIN {
            std::thread::sleep(Duration::from_secs(secs - FIVEMIN));
            let msg = "\x07\x07You have to leave in 5 minutes.\n";
            if write_stdout(msg) {
                process::exit(0);
            }
            secs = FIVEMIN;
        }

        if secs >= ONEMIN {
            std::thread::sleep(Duration::from_secs(secs - ONEMIN));
            let msg = "\x07\x07Just one more minute!\n";
            if write_stdout(msg) {
                process::exit(0);
            }
        }

        for _ in 0..10 {
            std::thread::sleep(Duration::from_secs(ONEMIN));
            let msg = "\x07\x07Time to leave!\n";
            if write_stdout(msg) {
                process::exit(0);
            }
        }

        let msg = "\x07\x07That was the last time I'll tell you.  Bye.\n";
        let _ = write_stdout(msg);
        process::exit(0);
    } else {
        eprintln!("fork failed");
        process::exit(1);
    }
}

fn write_stdout(msg: &str) -> bool {
    use std::os::unix::io::AsRawFd;
    let fd = io::stdout().as_raw_fd();
    unsafe {
        let ret = libc::write(fd, msg.as_ptr() as *const libc::c_void, msg.len());
        ret < 0
    }
}

fn chrono_like_format(dt: Duration) -> String {
    let secs = dt.as_secs();
    let yday = yday_from_secs(secs);
    let mut buf = [0u8; 80];
    unsafe {
        let tm = make_tm(secs);
        let fmt = b"%+\0";
        let ret = libc::strftime(
            buf.as_mut_ptr() as *mut libc::c_char,
            buf.len(),
            fmt.as_ptr() as *const libc::c_char,
            &tm,
        );
        if ret > 0 {
            String::from_utf8_lossy(&buf[..ret as usize]).into_owned()
        } else {
            "unknown".to_string()
        }
    }
}

fn yday_from_secs(_secs: u64) -> i32 {

}

fn make_tm(secs: u64) -> libc::tm {
    let mut tm: libc::tm = unsafe { std::mem::zeroed() };
    let mut ts = secs as libc::time_t;
    unsafe {
        libc::gmtime_r(&mut ts, &mut tm);
        // Convert to local time
        let mut local_tm: libc::tm = unsafe { std::mem::zeroed() };
        libc::localtime_r(&mut ts, &mut local_tm);
        local_tm
    }
}
